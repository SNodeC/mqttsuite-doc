93 MqttMapper::MqttMapper(const nlohmann::json& mappingJson)
94 : mappingJson(mappingJson)
95 , delayedQueue(this) {
96 injaEnvironment = new inja::Environment;
97
98 if (mappingJson.contains("plugins")) {
99 VLOG(1) << "Loading plugins ...";
100
101 for (const nlohmann::json& pluginJson : mappingJson["plugins"]) {
102 const std::string plugin = pluginJson;
103
104 void* handle = core::DynamicLoader::dlOpen(plugin);
105
106 if (handle != nullptr) {
107 pluginHandles.push_back(handle);
108
109 VLOG(1) << " Loading plugin: " << plugin << " ...";
110
111 const std::vector<mqtt::lib::Function>* loadedFunctions =
112 static_cast<std::vector<mqtt::lib::Function>*>(core::DynamicLoader::dlSym(handle, "functions"));
113 if (loadedFunctions != nullptr) {
114 VLOG(1) << " Registering inja 'none void callbacks'";
115 for (const mqtt::lib::Function& function : *loadedFunctions) {
116 VLOG(1) << " " << function.name;
117
118 if (function.numArgs >= 0) {
119 injaEnvironment->add_callback(function.name, function.numArgs, function.function);
120 } else {
121 injaEnvironment->add_callback(function.name, function.function);
122 }
123 }
124 VLOG(1) << " Registering inja 'none void callbacks done'";
125 } else {
126 VLOG(1) << " No inja none 'void callbacks found' in plugin " << plugin;
127 }
128
129 const std::vector<mqtt::lib::VoidFunction>* loadedVoidFunctions =
130 static_cast<std::vector<mqtt::lib::VoidFunction>*>(core::DynamicLoader::dlSym(handle, "voidFunctions"));
131 if (loadedVoidFunctions != nullptr) {
132 VLOG(1) << " Registering inja 'void callbacks'";
133 for (const mqtt::lib::VoidFunction& voidFunction : *loadedVoidFunctions) {
134 VLOG(1) << " " << voidFunction.name;
135
136 if (voidFunction.numArgs >= 0) {
137 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.numArgs, voidFunction.function);
138 } else {
139 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.function);
140 }
141 }
142 VLOG(1) << " Registering inja 'void callbacks' done";
143 } else {
144 VLOG(1) << " No inja 'void callbacks' found in plugin " << plugin;
145 }
146
147 VLOG(1) << " Loading plugin done: " << plugin;
148 } else {
149 VLOG(1) << " Error loading plugin: " << plugin;
150 }
151 }
152
153 VLOG(1) << "Loading plugins done";
154 }

177 void MqttMapper::publishMappings(const iot::mqtt::packets::Publish& publish) {
178 if (!mappingJson.empty()) {
179 nlohmann::json matchingTopicLevel = findMatchingTopicLevel(mappingJson["topic_level"], publish.getTopic());
180
181 if (!matchingTopicLevel.empty()) {
182 const nlohmann::json& subscription = matchingTopicLevel["subscription"];
183
184 if (subscription.contains("static")) {
185 VLOG(1) << "Topic mapping found for:";
186 VLOG(1) << " Type: static";
187 VLOG(1) << " Topic: " << publish.getTopic();
188 VLOG(1) << " Message: " << publish.getMessage();
189 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
190 VLOG(1) << " Retain: " << publish.getRetain();
191
192 publishMappedMessages(subscription["static"], publish);
193 }
194
195 if (subscription.contains("value")) {
196 VLOG(1) << "Topic mapping found for:";
197 VLOG(1) << " Type: value";
198 VLOG(1) << " Topic: " << publish.getTopic();
199 VLOG(1) << " Message: " << publish.getMessage();
200 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
201 VLOG(1) << " Retain: " << publish.getRetain();
202
203 nlohmann::json json;
204 json["message"] = publish.getMessage();
205
206 publishMappedTemplates(subscription["value"], json, publish);
207 }
208
209 if (subscription.contains("json")) {
210 VLOG(1) << "Topic mapping found for:";
211 VLOG(1) << " Type: json";
212 VLOG(1) << " Topic: " << publish.getTopic();
213 VLOG(1) << " Message: " << publish.getMessage();
214 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
215 VLOG(1) << " Retain: " << publish.getRetain();
216
217 try {
218 nlohmann::json json;
219 json["message"] = nlohmann::json::parse(publish.getMessage());
220
221 publishMappedTemplates(subscription["json"], json, publish);
222 } catch (const nlohmann::json::parse_error& e) {
223 VLOG(1) << " Parsing message into json failed: " << publish.getMessage();
224 VLOG(1) << " What: " << e.what() << '\n'
225 << " Exception Id: " << e.id << '\n'
226 << " Byte position of error: " << e.byte;
227 }
228 }
229 }
230 }

241 void MqttMapper::extractSubscription(const nlohmann::json& topicLevelJson,
242 const std::string& topic,
243 std::list<iot::mqtt::Topic>& topicList) {
244 const std::string name = topicLevelJson["name"];
245
246 if (topicLevelJson.contains("subscription")) {
247 const uint8_t qoS = topicLevelJson["subscription"]["qos"];
248
249 topicList.emplace_back(topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, qoS);
250 }
251
252 if (topicLevelJson.contains("topic_level")) {
253 extractSubscriptions(topicLevelJson, topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, topicList);
254 }
255 }

270 nlohmann::json MqttMapper::findMatchingTopicLevel(const nlohmann::json& topicLevel, const std::string& topic) {
271 nlohmann::json foundTopicLevel;
272
273 if (topicLevel.is_object()) {
274 const std::string::size_type slashPosition = topic.find('/');
275 const std::string topicLevelName = topic.substr(0, slashPosition);
276
277 if (topicLevel["name"] == topicLevelName || topicLevel["name"] == "+" || topicLevel["name"] == "#") {
278 if (slashPosition == std::string::npos) {
279 foundTopicLevel = topicLevel;
280 } else if (topicLevel.contains("topic_level")) {
281 foundTopicLevel = findMatchingTopicLevel(topicLevel["topic_level"], topic.substr(slashPosition + 1));
282 }
283 }
284 } else if (topicLevel.is_array()) {
285 for (const nlohmann::json& topicLevelEntry : topicLevel) {
286 foundTopicLevel = findMatchingTopicLevel(topicLevelEntry, topic);
287
288 if (!foundTopicLevel.empty()) {
289 break;
290 }
291 }
292 }
293
294 return foundTopicLevel;
295 }

297 void MqttMapper::publishMappedTemplate(const nlohmann::json& templateMapping, nlohmann::json& json) {
298 const std::string& mappingTemplate = templateMapping["mapping_template"];
299 const std::string& mappedTopic = templateMapping["mapped_topic"];
300
301 try {
302 // Render topic
303 const std::string renderedTopic = injaEnvironment->render(mappedTopic, json);
304 json["mapped_topic"] = renderedTopic;
305
306 VLOG(1) << " Mapped topic template: " << mappedTopic;
307 VLOG(1) << " -> " << renderedTopic;
308
309 try {
310 // Render message
311 const std::string renderedMessage = injaEnvironment->render(mappingTemplate, json);
312 VLOG(1) << " Mapped message template: " << mappingTemplate;
313 VLOG(1) << " -> " << renderedMessage;
314
315 const nlohmann::json& suppressions = templateMapping["suppressions"];
316 const bool retain = templateMapping["retain"];
317
318 if (suppressions.empty() || std::find(suppressions.begin(), suppressions.end(), renderedMessage) == suppressions.end() ||
319 (retain && renderedMessage.empty())) {
320 const uint8_t qoS = templateMapping["qos"];
321 const double delay = templateMapping["delay"];
322
323 VLOG(1) << " Send mapping:" << (delay > 0 ? " delayed" : "");
324 VLOG(1) << " Topic: " << renderedTopic;
325 VLOG(1) << " Message: " << renderedMessage << "";
326 VLOG(1) << " QoS: " << static_cast<int>(qoS);
327 VLOG(1) << " retain: " << retain;
328 VLOG(1) << " Delay: " << delay;
329
330 publishMapping(renderedTopic, renderedMessage, qoS, retain, delay);
331 } else {
332 VLOG(1) << " Rendered message: '" << renderedMessage << "' in suppression list:";
333 for (const nlohmann::json& item : suppressions) {
334 VLOG(1) << " '" << item.get<std::string>() << "'";
335 }
336 VLOG(1) << " Send mapping: suppressed";
337 }
338 } catch (const inja::InjaError& e) {
339 VLOG(1) << " Message template rendering failed: " << mappingTemplate << " : " << json.dump();
340 VLOG(1) << " What: " << e.what();
341 VLOG(1) << " INJA: " << e.type << ": " << e.message;
342 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
343 }
344 } catch (const inja::InjaError& e) {
345 VLOG(1) << " Topic template rendering failed: " << mappingTemplate << " : " << json.dump();
346 VLOG(1) << " What: " << e.what();
347 VLOG(1) << " INJA: " << e.type << ": " << e.message;
348 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
349 }
350 }

390 void MqttMapper::publishMappedMessage(const nlohmann::json& staticMapping, const iot::mqtt::packets::Publish& publish) {
391 const nlohmann::json& messageMapping = staticMapping["message_mapping"];
392
393 VLOG(1) << " Message mapping: " << messageMapping.dump();
394
395 if (messageMapping.is_object()) {
396 if (messageMapping["message"] == publish.getMessage()) {
397 publishMappedMessage(staticMapping["mapped_topic"],
398 messageMapping["mapped_message"],
399 staticMapping["qos"],
400 staticMapping["retain"],
401 staticMapping["delay"]);
402 } else {
403 VLOG(1) << " no matching mapped message found";
404 }
405 } else {
406 const nlohmann::json::const_iterator matchedMessageMappingIterator =
407 std::find_if(messageMapping.begin(), messageMapping.end(), [&publish](const nlohmann::json& messageMappingCandidat) {
408 return messageMappingCandidat["message"] == publish.getMessage();
409 });
410
411 if (matchedMessageMappingIterator != messageMapping.end()) {
412 publishMappedMessage(staticMapping["mapped_topic"],
413 (*matchedMessageMappingIterator)["mapped_message"],
414 staticMapping["qos"],
415 staticMapping["retain"],
416 staticMapping["delay"]);
417 } else {
418 VLOG(1) << " no matching mapped message found";
419 }
420 }
421 }

441 void MqttMapper::DelayedQueue::processDue() {
442 const auto now = utils::Timeval::currentTime();
443
444 while (!empty() && top().when <= now) {
445 const auto& sheduledPublish = top();
446 VLOG(1) << "Publish delayed message";
447 VLOG(1) << " Topic: " << sheduledPublish.topic;
448 VLOG(1) << " Message: " << sheduledPublish.message;
449 VLOG(1) << " QoS: " << static_cast<int>(sheduledPublish.qoS);
450 VLOG(1) << " retain: " << sheduledPublish.retain;
451 VLOG(1) << " Delay: " << sheduledPublish.delay;
452
453 mqttMapper->publishMapping(sheduledPublish.topic, sheduledPublish.message, sheduledPublish.qoS, sheduledPublish.retain);
454 pop();
455 }

458 void MqttMapper::DelayedQueue::armDelayTimer() {
459 delayTimer.cancel();
460
461 auto delay = top().when - utils::Timeval::currentTime();
462
463 // clamp negative delay to 0 (if top().when is already due)
464 if (delay < utils::Timeval{}) {
465 delay = utils::Timeval{};
466 }
467
468 delayTimer = core::timer::Timer::singleshotTimer(
469 [this]() {
470 processDue();
471
472 if (!empty()) {
473 armDelayTimer(); // re-arm for the next item
474 }
475 },
476 delay);
477 }

503 bool MqttMapper::EarlierFirst::operator()(const ScheduledPublish& a, const ScheduledPublish& b) const {
