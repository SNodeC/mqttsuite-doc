114 bool MqttMapper::setMapping(nlohmann::json mappingJson) { // can throw
115 delete injaEnvironment;
116
117 for (void* handle : pluginHandles) {
118 core::DynamicLoader::dlClose(handle);
119 }
120 pluginHandles.clear();
121
122 injaEnvironment = new inja::Environment;
123
124 nlohmann::json defaultPatch;
125
126 try {
127 defaultPatch = validator.validate(mappingJson);
128 } catch (const std::exception& e) {
129 throw std::runtime_error("Validating JSON failed: Mapping JSON = " + mappingJson.dump(4) + "\n" + e.what());
130 }
131
132 try {
133 mappingJson = mappingJson.patch(defaultPatch);
134 } catch (const std::exception& e) {
135 throw std::runtime_error("Patching JSON with default patch failed: Default patch = " + defaultPatch.dump(4) + "\n" + e.what());
136 }
137
138 bool mustReconnect = this->mappingJson["connection"] != mappingJson["connection"];
139
140 this->mappingJson = mappingJson;
141
142 if (mappingJson["mapping"].contains("plugins")) {
143 for (const nlohmann::json& pluginJson : mappingJson["mapping"]["plugins"]) {
144 const std::string plugin = pluginJson;
145
146 void* handle = core::DynamicLoader::dlOpen(plugin);
147
148 if (handle != nullptr) {
149 pluginHandles.push_back(handle);
150
151 VLOG(1) << " Loading plugin: " << plugin << " ...";
152
153 const std::vector<mqtt::lib::Function>* loadedFunctions =
154 static_cast<std::vector<mqtt::lib::Function>*>(core::DynamicLoader::dlSym(handle, "functions"));
155 if (loadedFunctions != nullptr) {
156 VLOG(1) << " Registering inja 'none void callbacks'";
157 for (const mqtt::lib::Function& function : *loadedFunctions) {
158 VLOG(1) << " " << function.name;
159
160 if (function.numArgs >= 0) {
161 injaEnvironment->add_callback(function.name, function.numArgs, function.function);
162 } else {
163 injaEnvironment->add_callback(function.name, function.function);
164 }
165 }
166 VLOG(1) << " Registering inja 'none void callbacks done'";
167 } else {
168 VLOG(1) << " No inja none 'void callbacks found' in plugin " << plugin;
169 }
170
171 const std::vector<mqtt::lib::VoidFunction>* loadedVoidFunctions =
172 static_cast<std::vector<mqtt::lib::VoidFunction>*>(core::DynamicLoader::dlSym(handle, "voidFunctions"));
173 if (loadedVoidFunctions != nullptr) {
174 VLOG(1) << " Registering inja 'void callbacks'";
175 for (const mqtt::lib::VoidFunction& voidFunction : *loadedVoidFunctions) {
176 VLOG(1) << " " << voidFunction.name;
177
178 if (voidFunction.numArgs >= 0) {
179 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.numArgs, voidFunction.function);
180 } else {
181 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.function);
182 }
183 }
184 VLOG(1) << " Registering inja 'void callbacks' done";
185 } else {
186 VLOG(1) << " No inja 'void callbacks' found in plugin " << plugin;
187 }
188
189 VLOG(1) << " Loading plugin done: " << plugin;
190 } else {
191 VLOG(1) << " Error loading plugin: " << plugin;
192 throw std::runtime_error("Error loading plugin '" + plugin + "': " + core::DynamicLoader::dlError());
193 }
194 }
195
196 VLOG(1) << "Loading plugins done";
197 }
198
199 return mustReconnect;
200 }

230 MqttMapper::MappedPublishes MqttMapper::getMappings(const iot::mqtt::packets::Publish& publish) {
231 MappedPublishes mappedPublishes;
232 if (mappingJson.contains("mapping") && !mappingJson["mapping"].empty()) {
233 nlohmann::json matchingTopicLevel = findMatchingTopicLevel(mappingJson["mapping"]["topic_level"], publish.getTopic());
234
235 if (!matchingTopicLevel.empty()) {
236 const nlohmann::json& subscription = matchingTopicLevel["subscription"];
237
238 if (subscription.contains("static")) {
239 VLOG(1) << "Topic mapping found for:";
240 VLOG(1) << " Type: static";
241 VLOG(1) << " Topic: " << publish.getTopic();
242 VLOG(1) << " Message: " << publish.getMessage();
243 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
244 VLOG(1) << " Retain: " << publish.getRetain();
245
246 getStaticMappings(subscription["static"], publish, mappedPublishes);
247 }
248
249 if (subscription.contains("value")) {
250 VLOG(1) << "Topic mapping found for:";
251 VLOG(1) << " Type: value";
252 VLOG(1) << " Topic: " << publish.getTopic();
253 VLOG(1) << " Message: " << publish.getMessage();
254 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
255 VLOG(1) << " Retain: " << publish.getRetain();
256
257 nlohmann::json json;
258 json["message"] = publish.getMessage();
259
260 getTemplateMappings(subscription["value"], json, publish, mappedPublishes);
261 }
262
263 if (subscription.contains("json")) {
264 VLOG(1) << "Topic mapping found for:";
265 VLOG(1) << " Type: json";
266 VLOG(1) << " Topic: " << publish.getTopic();
267 VLOG(1) << " Message: " << publish.getMessage();
268 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
269 VLOG(1) << " Retain: " << publish.getRetain();
270
271 try {
272 nlohmann::json json;
273 json["message"] = nlohmann::json::parse(publish.getMessage());
274
275 getTemplateMappings(subscription["json"], json, publish, mappedPublishes);
276 } catch (const nlohmann::json::parse_error& e) {
277 VLOG(1) << " Parsing message into json failed: " << publish.getMessage();
278 VLOG(1) << " What: " << e.what() << '\n'
279 << " Exception Id: " << e.id << '\n'
280 << " Byte position of error: " << e.byte;
281 }
282 }
283 }
284 }
285
286 return mappedPublishes;
287 }

289 void MqttMapper::extractSubscription(const nlohmann::json& topicLevelJson,
290 const std::string& topic,
291 std::list<iot::mqtt::Topic>& topicList) {
292 const std::string name = topicLevelJson["name"];
293
294 if (topicLevelJson.contains("subscription")) {
295 const uint8_t qoS = topicLevelJson["subscription"]["qos"];
296
297 topicList.emplace_back(topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, qoS);
298 }
299
300 if (topicLevelJson.contains("topic_level")) {
301 extractSubscriptions(topicLevelJson, topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, topicList);
302 }

320 nlohmann::json MqttMapper::findMatchingTopicLevel(const nlohmann::json& topicLevel, const std::string& topic) {
321 nlohmann::json foundTopicLevel;
322
323 if (topicLevel.is_object()) {
324 const std::string::size_type slashPosition = topic.find('/');
325 const std::string topicLevelName = topic.substr(0, slashPosition);
326
327 if (topicLevel["name"] == topicLevelName || topicLevel["name"] == "+" || topicLevel["name"] == "#") {
328 if (slashPosition == std::string::npos) {
329 foundTopicLevel = topicLevel;
330 } else if (topicLevel.contains("topic_level")) {
331 foundTopicLevel = findMatchingTopicLevel(topicLevel["topic_level"], topic.substr(slashPosition + 1));
332 }
333 }
334 } else if (topicLevel.is_array()) {
335 for (const nlohmann::json& topicLevelEntry : topicLevel) {
336 foundTopicLevel = findMatchingTopicLevel(topicLevelEntry, topic);
337
338 if (!foundTopicLevel.empty()) {
339 break;
340 }
341 }
342 }
343
344 return foundTopicLevel;
345 }

347 void MqttMapper::publishMappedTemplate(const nlohmann::json& templateMapping, nlohmann::json& json, MappedPublishes& mappedPublishes) {
348 const std::string& mappingTemplate = templateMapping["mapping_template"];
349 const std::string& mappedTopic = templateMapping["mapped_topic"];
350
351 try {
352 // Render topic
353 const std::string renderedTopic = injaEnvironment->render(mappedTopic, json);
354 json["mapped_topic"] = renderedTopic;
355
356 VLOG(1) << " Mapped topic template: " << mappedTopic;
357 VLOG(1) << " -> " << renderedTopic;
358
359 try {
360 // Render message
361 const std::string renderedMessage = injaEnvironment->render(mappingTemplate, json);
362 VLOG(1) << " Mapped message template: " << mappingTemplate;
363 VLOG(1) << " -> " << renderedMessage;
364
365 const nlohmann::json& suppressions = templateMapping["suppressions"];
366 const bool retain = templateMapping["retain"];
367
368 if (suppressions.empty() || std::find(suppressions.begin(), suppressions.end(), renderedMessage) == suppressions.end() ||
369 (retain && renderedMessage.empty())) {
370 const uint8_t qoS = templateMapping["qos"];
371 const double delay = templateMapping["delay"];
372
373 VLOG(1) << " Send mapping:" << (delay > 0 ? " delayed" : "");
374 VLOG(1) << " Topic: " << renderedTopic;
375 VLOG(1) << " Message: " << renderedMessage << "";
376 VLOG(1) << " QoS: " << static_cast<int>(qoS);
377 VLOG(1) << " retain: " << retain;
378 VLOG(1) << " Delay: " << delay;
379
380 publishMappedMessage(renderedTopic, renderedMessage, qoS, retain, delay, mappedPublishes);
381 } else {
382 VLOG(1) << " Rendered message: '" << renderedMessage << "' in suppression list:";
383 for (const nlohmann::json& item : suppressions) {
384 VLOG(1) << " '" << item.get<std::string>() << "'";
385 }
386 VLOG(1) << " Send mapping: suppressed";
387 }
388 } catch (const inja::InjaError& e) {
389 VLOG(1) << " Message template rendering failed: " << mappingTemplate << " : " << json.dump();
390 VLOG(1) << " What: " << e.what();
391 VLOG(1) << " INJA: " << e.type << ": " << e.message;
392 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
393 }
394 } catch (const inja::InjaError& e) {
395 VLOG(1) << " Topic template rendering failed: " << mappingTemplate << " : " << json.dump();
396 VLOG(1) << " What: " << e.what();
397 VLOG(1) << " INJA: " << e.type << ": " << e.message;
398 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
399 }

402 void MqttMapper::getTemplateMappings(const nlohmann::json& templateMapping,
403 nlohmann::json& json,
404 const iot::mqtt::packets::Publish& publish,
405 MappedPublishes& mappedPublishes) {
406 json["topic"] = publish.getTopic();
407 json["qos"] = publish.getQoS();
408 json["retain"] = publish.getRetain();
409 json["package_identifier"] = publish.getPacketIdentifier();
410
411 try {
412 VLOG(1) << " Render data: " << json.dump();
413
414 if (templateMapping.is_object()) {
415 publishMappedTemplate(templateMapping, json, mappedPublishes);
416 } else {
417 for (const nlohmann::json& concreteTemplateMapping : templateMapping) {
418 publishMappedTemplate(concreteTemplateMapping, json, mappedPublishes);
419 }
420 }
421 } catch (const nlohmann::json::exception& e) {
422 VLOG(1) << "JSON Exception during Render data:\n" << e.what();
423 }

426 void MqttMapper::publishMappedMessage(
427 const std::string& topic, const std::string& message, uint8_t qoS, bool retain, double delay, MappedPublishes& mappedPublishes) {
428 VLOG(1) << " Mapped topic:";
429 VLOG(1) << " -> " << topic;
430 VLOG(1) << " Mapped message:";
431 VLOG(1) << " -> " << message;
432 VLOG(1) << " Send mapping:" << (delay > 0 ? " delayed" : "");
433 VLOG(1) << " Topic: " << topic;
434 VLOG(1) << " Message: " << message;
435 VLOG(1) << " QoS: " << static_cast<int>(qoS);
436 VLOG(1) << " retain: " << retain;
437 VLOG(1) << " Delay: " << delay;
438
439 if (delay < 0.0) {
440 mappedPublishes.first.emplace_back(0, topic, message, qoS, false, retain);
441 } else {
442 mappedPublishes.second.push_back({delay, iot::mqtt::packets::Publish(0, topic, message, qoS, false, retain)});
443 }

446 void MqttMapper::publishMappedMessage(const nlohmann::json& staticMapping,
447 const iot::mqtt::packets::Publish& publish,
448 MappedPublishes& mappedPublishes) {
449 const nlohmann::json& messageMapping = staticMapping["message_mapping"];
450
451 VLOG(1) << " Message mapping: " << messageMapping.dump();
452
453 if (messageMapping.is_object()) {
454 if (messageMapping["message"] == publish.getMessage()) {
455 publishMappedMessage(staticMapping["mapped_topic"],
456 messageMapping["mapped_message"],
457 staticMapping["qos"],
458 staticMapping["retain"],
459 staticMapping["delay"],
460 mappedPublishes);
461 } else {
462 VLOG(1) << " no matching mapped message found";
463 }
464 } else {
465 const nlohmann::json::const_iterator matchedMessageMappingIterator =
466 std::find_if(messageMapping.begin(), messageMapping.end(), [&publish](const nlohmann::json& messageMappingCandidat) {
467 return messageMappingCandidat["message"] == publish.getMessage();
468 });
469
470 if (matchedMessageMappingIterator != messageMapping.end()) {
471 publishMappedMessage(staticMapping["mapped_topic"],
472 (*matchedMessageMappingIterator)["mapped_message"],
473 staticMapping["qos"],
474 staticMapping["retain"],
475 staticMapping["delay"],
476 mappedPublishes);
477 } else {
478 VLOG(1) << " no matching mapped message found";
479 }
480 }
