79 MqttMapper::MqttMapper(const nlohmann::json& mappingJson)
80 : mappingJson(mappingJson) {
81 injaEnvironment = new inja::Environment;
82
83 if (mappingJson.contains("plugins")) {
84 VLOG(1) << "Loading plugins ...";
85
86 for (const nlohmann::json& pluginJson : mappingJson["plugins"]) {
87 const std::string plugin = pluginJson;
88
89 void* handle = core::DynamicLoader::dlOpen(plugin);
90
91 if (handle != nullptr) {
92 pluginHandles.push_back(handle);
93
94 VLOG(1) << " Loading plugin: " << plugin << " ...";
95
96 const std::vector<mqtt::lib::Function>* loadedFunctions =
97 static_cast<std::vector<mqtt::lib::Function>*>(dlsym(handle, "functions"));
98 if (loadedFunctions != nullptr) {
99 VLOG(1) << " Registering inja 'none void callbacks'";
100 for (const mqtt::lib::Function& function : *loadedFunctions) {
101 VLOG(1) << " " << function.name;
102
103 if (function.numArgs >= 0) {
104 injaEnvironment->add_callback(function.name, function.numArgs, function.function);
105 } else {
106 injaEnvironment->add_callback(function.name, function.function);
107 }
108 }
109 VLOG(1) << " Registering inja 'none void callbacks done'";
110 } else {
111 VLOG(1) << " No inja none 'void callbacks found' in plugin " << plugin;
112 }
113
114 const std::vector<mqtt::lib::VoidFunction>* loadedVoidFunctions =
115 static_cast<std::vector<mqtt::lib::VoidFunction>*>(dlsym(handle, "voidFunctions"));
116 if (loadedVoidFunctions != nullptr) {
117 VLOG(1) << " Registering inja 'void callbacks'";
118 for (const mqtt::lib::VoidFunction& voidFunction : *loadedVoidFunctions) {
119 VLOG(1) << " " << voidFunction.name;
120
121 if (voidFunction.numArgs >= 0) {
122 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.numArgs, voidFunction.function);
123 } else {
124 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.function);
125 }
126 }
127 VLOG(1) << " Registering inja 'void callbacks' done";
128 } else {
129 VLOG(1) << " No inja 'void callbacks' found in plugin " << plugin;
130 }
131
132 VLOG(1) << " Loading plugin done: " << plugin;
133 } else {
134 VLOG(1) << " Error loading plugin: " << plugin;
135 }
136 }
137
138 VLOG(1) << "Loading plugins done";
139 }
140 }

162 void MqttMapper::publishMappings(const iot::mqtt::packets::Publish& publish) {
163 if (!mappingJson.empty()) {
164 nlohmann::json matchingTopicLevel = findMatchingTopicLevel(mappingJson["topic_level"], publish.getTopic());
165
166 if (!matchingTopicLevel.empty()) {
167 const nlohmann::json& subscription = matchingTopicLevel["subscription"];
168
169 if (subscription.contains("static")) {
170 VLOG(1) << "Topic mapping found for:";
171 VLOG(1) << " Type: static";
172 VLOG(1) << " Topic: " << publish.getTopic();
173 VLOG(1) << " Message: " << publish.getMessage();
174 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
175 VLOG(1) << " Retain: " << publish.getRetain();
176
177 publishMappedMessages(subscription["static"], publish);
178 }
179
180 if (subscription.contains("value")) {
181 VLOG(1) << "Topic mapping found for:";
182 VLOG(1) << " Type: value";
183 VLOG(1) << " Topic: " << publish.getTopic();
184 VLOG(1) << " Message: " << publish.getMessage();
185 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
186 VLOG(1) << " Retain: " << publish.getRetain();
187
188 nlohmann::json json;
189 json["message"] = publish.getMessage();
190
191 publishMappedTemplates(subscription["value"], json, publish);
192 }
193
194 if (subscription.contains("json")) {
195 VLOG(1) << "Topic mapping found for:";
196 VLOG(1) << " Type: json";
197 VLOG(1) << " Topic: " << publish.getTopic();
198 VLOG(1) << " Message: " << publish.getMessage();
199 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
200 VLOG(1) << " Retain: " << publish.getRetain();
201
202 try {
203 nlohmann::json json;
204 json["message"] = nlohmann::json::parse(publish.getMessage());
205
206 publishMappedTemplates(subscription["json"], json, publish);
207 } catch (const nlohmann::json::parse_error& e) {
208 VLOG(1) << " Parsing message into json failed: " << publish.getMessage();
209 VLOG(1) << " What: " << e.what() << '\n'
210 << " Exception Id: " << e.id << '\n'
211 << " Byte position of error: " << e.byte;
212 }
213 }
214 }
215 }

218 void MqttMapper::extractSubscription(const nlohmann::json& topicLevelJson,
219 const std::string& topic,
220 std::list<iot::mqtt::Topic>& topicList) {
221 const std::string name = topicLevelJson["name"];
222
223 if (topicLevelJson.contains("subscription")) {
224 const uint8_t qoS = topicLevelJson["subscription"]["qos"];
225
226 topicList.emplace_back(topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, qoS);
227 }
228
229 if (topicLevelJson.contains("topic_level")) {
230 extractSubscriptions(topicLevelJson, topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, topicList);
231 }
232 }

247 nlohmann::json MqttMapper::findMatchingTopicLevel(const nlohmann::json& topicLevel, const std::string& topic) {
248 nlohmann::json foundTopicLevel;
249
250 if (topicLevel.is_object()) {
251 const std::string::size_type slashPosition = topic.find('/');
252 const std::string topicLevelName = topic.substr(0, slashPosition);
253
254 if (topicLevel["name"] == topicLevelName) {
255 if (slashPosition == std::string::npos) {
256 foundTopicLevel = topicLevel;
257 } else if (topicLevel.contains("topic_level")) {
258 foundTopicLevel = findMatchingTopicLevel(topicLevel["topic_level"], topic.substr(slashPosition + 1));
259 }
260 }
261 } else if (topicLevel.is_array()) {
262 for (const nlohmann::json& topicLevelEntry : topicLevel) {
263 foundTopicLevel = findMatchingTopicLevel(topicLevelEntry, topic);
264
265 if (!foundTopicLevel.empty()) {
266 break;
267 }
268 }
269 }
270
271 return foundTopicLevel;
272 }

274 void MqttMapper::publishMappedTemplate(const nlohmann::json& templateMapping, nlohmann::json& json) {
275 const std::string& mappingTemplate = templateMapping["mapping_template"];
276 const std::string& mappedTopic = templateMapping["mapped_topic"];
277
278 try {
279 // Render topic
280 const std::string renderedTopic = injaEnvironment->render(mappedTopic, json);
281 json["mapped_topic"] = renderedTopic;
282
283 VLOG(1) << " Mapped topic template: " << mappedTopic;
284 VLOG(1) << " -> " << renderedTopic;
285
286 try {
287 // Render message
288 const std::string renderedMessage = injaEnvironment->render(mappingTemplate, json);
289 VLOG(1) << " Mapped message template: " << mappingTemplate;
290 VLOG(1) << " -> " << renderedMessage;
291
292 const nlohmann::json& suppressions = templateMapping["suppressions"];
293 const bool retain = templateMapping["retain"];
294
295 if (suppressions.empty() || std::find(suppressions.begin(), suppressions.end(), renderedMessage) == suppressions.end() ||
296 (retain && renderedMessage.empty())) {
297 const uint8_t qoS = templateMapping["qos"];
298
299 VLOG(1) << " Send mapping:";
300 VLOG(1) << " Topic: " << renderedTopic;
301 VLOG(1) << " Message: " << renderedMessage << "";
302 VLOG(1) << " QoS: " << static_cast<int>(qoS);
303 VLOG(1) << " retain: " << retain;
304
305 publishMapping(renderedTopic, renderedMessage, qoS, retain);
306 } else {
307 VLOG(1) << " Rendered message: '" << renderedMessage << "' in suppression list:";
308 for (const nlohmann::json& item : suppressions) {
309 VLOG(1) << " '" << item.get<std::string>() << "'";
310 }
311 VLOG(1) << " Send mapping: suppressed";
312 }
313 } catch (const inja::InjaError& e) {
314 VLOG(1) << " Message template rendering failed: " << mappingTemplate << " : " << json.dump();
315 VLOG(1) << " What: " << e.what();
316 VLOG(1) << " INJA: " << e.type << ": " << e.message;
317 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
318 }
319 } catch (const inja::InjaError& e) {
320 VLOG(1) << " Topic template rendering failed: " << mappingTemplate << " : " << json.dump();
321 VLOG(1) << " What: " << e.what();
322 VLOG(1) << " INJA: " << e.type << ": " << e.message;
323 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
324 }

327 void MqttMapper::publishMappedTemplates(const nlohmann::json& templateMapping,
328 nlohmann::json& json,
329 const iot::mqtt::packets::Publish& publish) {
330 json["topic"] = publish.getTopic();
331 json["qos"] = publish.getQoS();
332 json["retain"] = publish.getRetain();
333 json["package_identifier"] = publish.getPacketIdentifier();
334
335 try {
336 VLOG(1) << " Render data: " << json.dump();
337
338 if (templateMapping.is_object()) {
339 publishMappedTemplate(templateMapping, json);
340 } else {
341 for (const nlohmann::json& concreteTemplateMapping : templateMapping) {
342 publishMappedTemplate(concreteTemplateMapping, json);
343 }
344 }
345 } catch (const nlohmann::json::exception& e) {
346 VLOG(1) << "JSON Exception during Render data:\n" << e.what();
347 }

364 void MqttMapper::publishMappedMessage(const nlohmann::json& staticMapping, const iot::mqtt::packets::Publish& publish) {
365 const nlohmann::json& messageMapping = staticMapping["message_mapping"];
366
367 VLOG(1) << " Message mapping: " << messageMapping.dump();
368
369 if (messageMapping.is_object()) {
370 if (messageMapping["message"] == publish.getMessage()) {
371 publishMappedMessage(
372 staticMapping["mapped_topic"], messageMapping["mapped_message"], staticMapping["qos"], staticMapping["retain"]);
373 } else {
374 VLOG(1) << " no matching mapped message found";
375 }
376 } else {
377 const nlohmann::json::const_iterator matchedMessageMappingIterator =
378 std::find_if(messageMapping.begin(), messageMapping.end(), [&publish](const nlohmann::json& messageMappingCandidat) {
379 return messageMappingCandidat["message"] == publish.getMessage();
380 });
381
382 if (matchedMessageMappingIterator != messageMapping.end()) {
383 publishMappedMessage(staticMapping["mapped_topic"],
384 (*matchedMessageMappingIterator)["mapped_message"],
385 staticMapping["qos"],
386 staticMapping["retain"]);
387 } else {
388 VLOG(1) << " no matching mapped message found";
389 }
390 }
391 }
