57 MqttMapper::MqttMapper(const nlohmann::json& mappingJson)
58 : mappingJson(mappingJson) {
59 injaEnvironment = new inja::Environment;
60
61 if (mappingJson.contains("plugins")) {
62 VLOG(1) << "Loading plugins ...";
63
64 for (const nlohmann::json& pluginJson : mappingJson["plugins"]) {
65 const std::string plugin = pluginJson;
66
67 void* handle = core::DynamicLoader::dlOpen(plugin);
68
69 if (handle != nullptr) {
70 pluginHandles.push_back(handle);
71
72 VLOG(1) << " Loading plugin: " << plugin << " ...";
73
74 const std::vector<mqtt::lib::Function>* loadedFunctions =
75 static_cast<std::vector<mqtt::lib::Function>*>(dlsym(handle, "functions"));
76 if (loadedFunctions != nullptr) {
77 VLOG(1) << " Registering inja 'none void callbacks'";
78 for (const mqtt::lib::Function& function : *loadedFunctions) {
79 VLOG(1) << " " << function.name;
80
81 if (function.numArgs >= 0) {
82 injaEnvironment->add_callback(function.name, function.numArgs, function.function);
83 } else {
84 injaEnvironment->add_callback(function.name, function.function);
85 }
86 }
87 VLOG(1) << " Registering inja 'none void callbacks done'";
88 } else {
89 VLOG(1) << " No inja none 'void callbacks found' in plugin " << plugin;
90 }
91
92 const std::vector<mqtt::lib::VoidFunction>* loadedVoidFunctions =
93 static_cast<std::vector<mqtt::lib::VoidFunction>*>(dlsym(handle, "voidFunctions"));
94 if (loadedVoidFunctions != nullptr) {
95 VLOG(1) << " Registering inja 'void callbacks'";
96 for (const mqtt::lib::VoidFunction& voidFunction : *loadedVoidFunctions) {
97 VLOG(1) << " " << voidFunction.name;
98
99 if (voidFunction.numArgs >= 0) {
100 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.numArgs, voidFunction.function);
101 } else {
102 injaEnvironment->add_void_callback(voidFunction.name, voidFunction.function);
103 }
104 }
105 VLOG(1) << " Registering inja 'void callbacks' done";
106 } else {
107 VLOG(1) << " No inja 'void callbacks' found in plugin " << plugin;
108 }
109
110 VLOG(1) << " Loading plugin done: " << plugin;
111 } else {
112 VLOG(1) << " Error loading plugin: " << plugin;
113 }
114 }
115
116 VLOG(1) << "Loading plugins done";
117 }
118 }

140 void MqttMapper::publishMappings(const iot::mqtt::packets::Publish& publish) {
141 if (!mappingJson.empty()) {
142 nlohmann::json matchingTopicLevel = findMatchingTopicLevel(mappingJson["topic_level"], publish.getTopic());
143
144 if (!matchingTopicLevel.empty()) {
145 const nlohmann::json& subscription = matchingTopicLevel["subscription"];
146
147 if (subscription.contains("static")) {
148 VLOG(1) << "Topic mapping found for:";
149 VLOG(1) << " Type: static";
150 VLOG(1) << " Topic: " << publish.getTopic();
151 VLOG(1) << " Message: " << publish.getMessage();
152 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
153 VLOG(1) << " Retain: " << publish.getRetain();
154
155 publishMappedMessages(subscription["static"], publish);
156 }
157
158 if (subscription.contains("value")) {
159 VLOG(1) << "Topic mapping found for:";
160 VLOG(1) << " Type: value";
161 VLOG(1) << " Topic: " << publish.getTopic();
162 VLOG(1) << " Message: " << publish.getMessage();
163 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
164 VLOG(1) << " Retain: " << publish.getRetain();
165
166 nlohmann::json json;
167 json["message"] = publish.getMessage();
168
169 publishMappedTemplates(subscription["value"], json, publish);
170 }
171
172 if (subscription.contains("json")) {
173 VLOG(1) << "Topic mapping found for:";
174 VLOG(1) << " Type: json";
175 VLOG(1) << " Topic: " << publish.getTopic();
176 VLOG(1) << " Message: " << publish.getMessage();
177 VLOG(1) << " QoS: " << static_cast<uint16_t>(publish.getQoS());
178 VLOG(1) << " Retain: " << publish.getRetain();
179
180 try {
181 nlohmann::json json;
182 json["message"] = nlohmann::json::parse(publish.getMessage());
183
184 publishMappedTemplates(subscription["json"], json, publish);
185 } catch (const nlohmann::json::parse_error& e) {
186 VLOG(1) << " Parsing message into json failed: " << publish.getMessage();
187 VLOG(1) << " What: " << e.what() << '\n'
188 << " Exception Id: " << e.id << '\n'
189 << " Byte position of error: " << e.byte;
190 }
191 }
192 }
193 }

196 void MqttMapper::extractSubscription(const nlohmann::json& topicLevelJson,
197 const std::string& topic,
198 std::list<iot::mqtt::Topic>& topicList) {
199 const std::string name = topicLevelJson["name"];
200
201 if (topicLevelJson.contains("subscription")) {
202 const uint8_t qoS = topicLevelJson["subscription"]["qos"];
203
204 topicList.emplace_back(topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, qoS);
205 }
206
207 if (topicLevelJson.contains("topic_level")) {
208 extractSubscriptions(topicLevelJson, topic + ((topic.empty() || topic == "/") && !name.empty() ? "" : "/") + name, topicList);
209 }
210 }

225 nlohmann::json MqttMapper::findMatchingTopicLevel(const nlohmann::json& topicLevel, const std::string& topic) {
226 nlohmann::json foundTopicLevel;
227
228 if (topicLevel.is_object()) {
229 const std::string::size_type slashPosition = topic.find('/');
230 const std::string topicLevelName = topic.substr(0, slashPosition);
231
232 if (topicLevel["name"] == topicLevelName) {
233 if (slashPosition == std::string::npos) {
234 foundTopicLevel = topicLevel;
235 } else if (topicLevel.contains("topic_level")) {
236 foundTopicLevel = findMatchingTopicLevel(topicLevel["topic_level"], topic.substr(slashPosition + 1));
237 }
238 }
239 } else if (topicLevel.is_array()) {
240 for (const nlohmann::json& topicLevelEntry : topicLevel) {
241 foundTopicLevel = findMatchingTopicLevel(topicLevelEntry, topic);
242
243 if (!foundTopicLevel.empty()) {
244 break;
245 }
246 }
247 }
248
249 return foundTopicLevel;
250 }

252 void MqttMapper::publishMappedTemplate(const nlohmann::json& templateMapping, nlohmann::json& json) {
253 const std::string& mappingTemplate = templateMapping["mapping_template"];
254 const std::string& mappedTopic = templateMapping["mapped_topic"];
255
256 try {
257 // Render topic
258 const std::string renderedTopic = injaEnvironment->render(mappedTopic, json);
259 json["mapped_topic"] = renderedTopic;
260
261 VLOG(1) << " Mapped topic template: " << mappedTopic;
262 VLOG(1) << " -> " << renderedTopic;
263
264 try {
265 // Render message
266 const std::string renderedMessage = injaEnvironment->render(mappingTemplate, json);
267 VLOG(1) << " Mapped message template: " << mappingTemplate;
268 VLOG(1) << " -> " << renderedMessage;
269
270 const nlohmann::json& suppressions = templateMapping["suppressions"];
271 const bool retain = templateMapping["retain"];
272
273 if (suppressions.empty() || std::find(suppressions.begin(), suppressions.end(), renderedMessage) == suppressions.end() ||
274 (retain && renderedMessage.empty())) {
275 const uint8_t qoS = templateMapping["qos"];
276
277 VLOG(1) << " Send mapping:";
278 VLOG(1) << " Topic: " << renderedTopic;
279 VLOG(1) << " Message: " << renderedMessage << "";
280 VLOG(1) << " QoS: " << static_cast<int>(qoS);
281 VLOG(1) << " retain: " << retain;
282
283 publishMapping(renderedTopic, renderedMessage, qoS, retain);
284 } else {
285 VLOG(1) << " Rendered message: '" << renderedMessage << "' in suppression list:";
286 for (const nlohmann::json& item : suppressions) {
287 VLOG(1) << " '" << item.get<std::string>() << "'";
288 }
289 VLOG(1) << " Send mapping: suppressed";
290 }
291 } catch (const inja::InjaError& e) {
292 VLOG(1) << " Message template rendering failed: " << mappingTemplate << " : " << json.dump();
293 VLOG(1) << " What: " << e.what();
294 VLOG(1) << " INJA: " << e.type << ": " << e.message;
295 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
296 }
297 } catch (const inja::InjaError& e) {
298 VLOG(1) << " Topic template rendering failed: " << mappingTemplate << " : " << json.dump();
299 VLOG(1) << " What: " << e.what();
300 VLOG(1) << " INJA: " << e.type << ": " << e.message;
301 VLOG(1) << " INJA (line:column):" << e.location.line << ":" << e.location.column;
302 }

306 nlohmann::json& json,
307 const iot::mqtt::packets::Publish& publish) {
308 json["topic"] = publish.getTopic();
309 json["qos"] = publish.getQoS();
310 json["retain"] = publish.getRetain();
311 json["package_identifier"] = publish.getPacketIdentifier();
312
313 try {
314 VLOG(1) << " Render data: " << json.dump();
315
316 if (templateMapping.is_object()) {
317 publishMappedTemplate(templateMapping, json);
318 } else {
319 for (const nlohmann::json& concreteTemplateMapping : templateMapping) {
320 publishMappedTemplate(concreteTemplateMapping, json);
321 }
322 }
323 } catch (const nlohmann::json::exception& e) {
324 VLOG(1) << "JSON Exception during Render data:\n" << e.what();
325 }

342 void MqttMapper::publishMappedMessage(const nlohmann::json& staticMapping, const iot::mqtt::packets::Publish& publish) {
343 const nlohmann::json& messageMapping = staticMapping["message_mapping"];
344
345 VLOG(1) << " Message mapping: " << messageMapping.dump();
346
347 if (messageMapping.is_object()) {
348 if (messageMapping["message"] == publish.getMessage()) {
349 publishMappedMessage(
350 staticMapping["mapped_topic"], messageMapping["mapped_message"], staticMapping["qos"], staticMapping["retain"]);
351 } else {
352 VLOG(1) << " no matching mapped message found";
353 }
354 } else {
355 const nlohmann::json::const_iterator matchedMessageMappingIterator =
356 std::find_if(messageMapping.begin(), messageMapping.end(), [&publish](const nlohmann::json& messageMappingCandidat) {
357 return messageMappingCandidat["message"] == publish.getMessage();
358 });
359
360 if (matchedMessageMappingIterator != messageMapping.end()) {
361 publishMappedMessage(staticMapping["mapped_topic"],
362 (*matchedMessageMappingIterator)["mapped_message"],
363 staticMapping["qos"],
364 staticMapping["retain"]);
365 } else {
366 VLOG(1) << " no matching mapped message found";
367 }
368 }
369 }
