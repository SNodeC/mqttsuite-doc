80static void
81reportState(const std::string& instanceName, const core::socket::SocketAddress& socketAddress, const core::socket::State& state) {
82 switch (state) {
83 case core::socket::State::OK:
84 VLOG(1) << instanceName << ": connected to '" << socketAddress.toString() << "'";
85 break;
86 case core::socket::State::DISABLED:
87 VLOG(1) << instanceName << ": disabled";
88 break;
89 case core::socket::State::ERROR:
90 VLOG(1) << instanceName << ": " << socketAddress.toString() << ": " << state.what();
91 break;
92 case core::socket::State::FATAL:
93 VLOG(1) << instanceName << ": " << socketAddress.toString() << ": " << state.what();
94 break;
95 }
96}

98template <typename HttpClient>
99void startClient(const std::string& name, const std::function<void(typename HttpClient::Config&)>& configurator) {
100 using SocketAddress = typename HttpClient::SocketAddress;
101
102 const HttpClient httpClient(
103 name,
104 [](const std::shared_ptr<web::http::client::Request>& req) {
105 req->set("Sec-WebSocket-Protocol", "mqtt");
106
107 if (!req->upgrade(
108 "/ws",
109 "websocket",
110 [](const std::shared_ptr<web::http::client::Request>& req, const std::shared_ptr<web::http::client::Response>& res) {
111 req->upgrade(res, [subProtocolsRequested = req->header("Upgrade")](const std::string& name) {
112 if (!name.empty()) {
113 VLOG(1) << "Successful upgrade to '" << name << "' requested: " << subProtocolsRequested;
114 } else {
115 VLOG(1) << "Can not upgrade to any of '" << subProtocolsRequested << "'";
116 }
117 });
118 },
119 [](const std::shared_ptr<web::http::client::Request>& req, const std::string& reason) {
120 VLOG(1) << "Upgrade to subprotocols '" << req->header("Upgrade")
121 << "' failed with response parse error: " << reason;
122 })) {
123 VLOG(1) << "Initiating upgrade to any of 'upgradeprotocol, websocket' failed";
124 }
125 },
126 []([[maybe_unused]] const std::shared_ptr<web::http::client::Request>& req) {
127 VLOG(1) << "Session ended";
128 });
129
130 configurator(httpClient.getConfig());
131
132 httpClient.connect([name](const SocketAddress& socketAddress, const core::socket::State& state) {
133 reportState(name, socketAddress, state);
134 });
135}
