66 express::Router makeMappingAdminRouter(const std::string& mappingFilePath, const AdminOptions& opt, ReloadCallback onDeploy) {
67 express::Router api;
68 const auto& schema = JsonMappingReader::getSchema();
69 auto validator =
70 std::make_shared<nlohmann::json_schema::json_validator>(schema, nullptr, nlohmann::json_schema::default_string_format_check);
71
72 api.use(express::middleware::JsonMiddleware());
73 api.use(express::middleware::BasicAuthentication(opt.user, opt.pass, opt.realm));
74
75 // GET /schema
76 api.get("/schema", [schema] APPLICATION(req, res) {
77 res->status(200).json(schema);
78 });
79
80 // GET /config
81 api.get("/config", [mappingFilePath] APPLICATION(req, res) {
82 try {
83 res->status(200).json(JsonMappingReader::readDraftOrActive(mappingFilePath));
84 } catch (const std::exception& e) {
85 res->status(500).json({{"error", "Failed to load configuration"}, {"details", e.what()}});
86 }
87 });
88
89 // PATCH /config
90 api.patch("/config", [mappingFilePath] APPLICATION(req, res) {
91 try {
92 const std::string bodyStr(req->body.begin(), req->body.end());
93 json patchOps = json::parse(bodyStr);
94
95 json current = JsonMappingReader::readDraftOrActive(mappingFilePath);
96 current = current.patch(patchOps);
97
98 JsonMappingReader::saveDraft(mappingFilePath, current);
99
100 res->status(200).json({{"status", "patched"}, {"path", mappingFilePath}});
101 } catch (const json::parse_error& e) {
102 res->status(400).json({{"error", "Invalid JSON body"}, {"details", e.what()}});
103 } catch (const std::exception& e) {
104 res->status(422).json({{"error", "Patch application failed"}, {"details", e.what()}});
105 }
106 });
107
108 // POST /config/deploy
109 api.post("/config/deploy", [mappingFilePath, onDeploy] APPLICATION(req, res) {
110 try {
111 JsonMappingReader::deployDraft(mappingFilePath);
112 if (onDeploy)
113 onDeploy();
114 res->status(200).json({{"status", "deploy-ack"}, {"note", "hot-reload triggered"}});
115 } catch (const std::exception& e) {
116 res->status(500).json({{"error", "Deploy failed"}, {"details", e.what()}});
117 }
118 });
119
120 // POST /config/validate
121 api.post("/config/validate", [validator] APPLICATION(req, res) {
122 try {
123 const std::string bodyStr(req->body.begin(), req->body.end());
124 auto document = nlohmann::json::parse(bodyStr);
125
126 nlohmann::json_schema::basic_error_handler err;
127 validator->validate(document, err);
128
129 if (err) {
130 res->status(422).json({{"valid", false}, {"error", "Validation failed"}});
131 } else {
132 res->status(200).json({{"valid", true}});
133 }
134 } catch (const std::exception& e) {
135 res->status(400).json({{"error", "Validation exception"}, {"details", e.what()}});
136 }
137 });
138
139 // POST /config/rollback
140 api.post("/config/rollback", [mappingFilePath, onDeploy] APPLICATION(req, res) {
141 try {
142 const std::string bodyStr(req->body.begin(), req->body.end());
143 auto jsonBody = nlohmann::json::parse(bodyStr);
144
145 if (!jsonBody.contains("version_id")) {
146 res->status(400).json({{"error", "Missing version_id"}});
147 return;
148 }
149
150 std::string versionId = jsonBody["version_id"];
151 JsonMappingReader::rollbackTo(mappingFilePath, versionId);
152
153 if (onDeploy)
154 onDeploy(); // Trigger hot-reload
155
156 res->status(200).json({{"status", "rolled_back"}, {"version", versionId}});
157 } catch (const std::exception& e) {
158 res->status(500).json({{"error", "Rollback failed"}, {"details", e.what()}});
159 }
160 });
161
162 // GET /config/history
163 api.get("/config/history", [mappingFilePath] APPLICATION(req, res) {
164 try {
165 auto history = JsonMappingReader::getHistory(mappingFilePath);
166 nlohmann::json list = nlohmann::json::array();
167 for (const auto& h : history) {
168 list.push_back({{"id", h.id}, {"comment", h.comment}, {"date", h.date}});
169 }
170 res->status(200).json(list);
171 } catch ([[maybe_unused]] const std::exception& e) {
172 res->status(500).json({{"error", "Failed to fetch history"}});
173 }
174 });
175
176 api.get("/", [] APPLICATION(req, res) {
177 res->redirect("/ui");
178 });
179
180 api.get("/ui", [] APPLICATION(req, res) {
181 res->redirect("/ui/index.html");
182 });
183
184 api.use("/ui",
185 express::middleware::StaticMiddleware("/home/voc/tmp/integrator/mqtt-integrator-ui/dist/mqtt-integrator-ui/browser"));
186
187 api.get("*", [] APPLICATION(req, res) {
188 res->redirect("/ui/index.html");
189 });
190
191 return api;
192 }
