55 ConfigSubscribe::ConfigSubscribe(net::config::ConfigInstance* instance)
56 : net::config::ConfigSection(instance, this, "Applications (at least one required)") {
57 topicOpt = addOptionFunction<std::string>(
58 "--topic",
59 [configSubscribe = this](const std::string& value) {
60 if (!value.empty()) {
61 configSubscribe->required(configSubscribe->topicOpt, false);
62
63 if (configSubscribe->configPublish != nullptr && !configSubscribe->getRequired()) {
64 configSubscribe->configPublish->required(false);
65 }
66 }
67 },
68 "List of topics subscribing to")
69 ->group(sectionSc->get_formatter()->get_label("Persistent Options"))
70 ->type_name("string list")
71 ->take_all()
72 ->allow_extra_args()
73 ->configurable();
74 sectionSc->needs(topicOpt);
75
76 required(topicOpt);
77
78 required(false);
79 }
