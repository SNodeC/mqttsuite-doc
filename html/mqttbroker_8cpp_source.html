95static void upgrade APPLICATION(req, res) {
96 if (req->get("sec-websocket-protocol").find("mqtt") != std::string::npos) {
97 res->upgrade(req, [req, res](const std::string& name) {
98 if (!name.empty()) {
99 VLOG(1) << "Successful upgrade to '" << name << "' requested: " << req->get("upgrade");
100 } else {
101 VLOG(1) << "Can not upgrade to any of '" << req->get("upgrade") << "'";
102 }
103 res->end();
104 });
105 } else {
106 VLOG(1) << "Not supporting any of: " << req->get("sec-websocket-protocol");
107
108 res->sendStatus(404);
109 }
110}

206static std::string urlDecode(const std::string& encoded) {
207 std::string decoded;
208 size_t i = 0;
209
210 while (i < encoded.length()) {
211 char ch = encoded[i];
212 if (ch == '%') {
213 // Make sure there are at least two characters after '%'
214 if (i + 2 < encoded.length() && std::isxdigit(encoded[i + 1]) && std::isxdigit(encoded[i + 2])) {
215 // Convert the two hex digits to a character
216 std::string hexValue = encoded.substr(i + 1, 2);
217 char decodedChar = static_cast<char>(std::stoi(hexValue, nullptr, 16));
218 decoded.push_back(decodedChar);
219 i += 3; // Skip over the % and the two hex digits
220 } else {
221 // Malformed encoding, just add the '%' as is.
222 decoded.push_back(ch);
223 ++i;
224 }
225 } else if (ch == '+') {
226 // Convert '+' to space (common in URL encoding)
227 decoded.push_back(' ');
228 ++i;
229 } else {
230 // Regular character, just append it.
231 decoded.push_back(ch);
232 ++i;
233 }
234 }
235
236 return decoded;
237}

239static express::Router getRouter(inja::Environment& environment) {
240 const express::Router router;
241
242 const express::Router& jsonRouter = express::middleware::JsonMiddleware();
243
244 jsonRouter.post("/clients", [] APPLICATION(req, res) {
245 req->getAttribute<nlohmann::json>(
246 [&res](nlohmann::json& json) {
247 std::string jsonString = json.dump(4);
248 VLOG(1) << "Application received JSON body\n" << jsonString;
249
250 std::string clientId = json["client_id"].get<std::string>();
251 const mqtt::mqttbroker::lib::Mqtt* mqtt = mqtt::mqttbroker::lib::MqttModel::instance().getMqtt(clientId);
252
253 if (mqtt != nullptr) {
254 mqtt->getMqttContext()->getSocketConnection()->close();
255 res->send(jsonString);
256 } else {
257 res->status(404).send("MQTT client has already gone away: " + json["connection_name"].get<std::string>());
258 }
259 },
260 [&res](const std::string& key) {
261 VLOG(1) << "Attribute type not found: " << key;
262
263 res->status(400).send("Attribute type not found: " + key);
264 });
265 });
266
267 jsonRouter.post("/unsubscribe", [] APPLICATION(req, res) {
268 req->getAttribute<nlohmann::json>(
269 [&res](nlohmann::json& json) {
270 std::string jsonString = json.dump(4);
271 VLOG(1) << "Application received JSON body\n" << jsonString;
272
273 std::string clientId = json["client_id"].get<std::string>();
274 std::string topic = json["topic"].get<std::string>();
275
276 const mqtt::mqttbroker::lib::Mqtt* mqtt = mqtt::mqttbroker::lib::MqttModel::instance().getMqtt(clientId);
277
278 if (mqtt != nullptr) {
279 mqtt->unsubscribe(topic);
280 res->send(jsonString);
281 } else {
282 res->status(404).send("MQTT client has already gone away: " + json["connection_name"].get<std::string>());
283 }
284 },
285 [&res](const std::string& key) {
286 VLOG(1) << "Attribute type not found: " << key;
287
288 res->status(400).send("Attribute type not found: " + key);
289 });
290 });
291
292 jsonRouter.post("/subscribe", [] APPLICATION(req, res) {
293 req->getAttribute<nlohmann::json>(
294 [&res](nlohmann::json& json) {
295 std::string jsonString = json.dump(4);
296 VLOG(1) << "Application received JSON body\n" << jsonString;
297
298 std::string clientId = json["client_id"].get<std::string>();
299 std::string topic = json["topic"].get<std::string>();
300 uint8_t qoS = json["qos"].get<uint8_t>();
301
302 const mqtt::mqttbroker::lib::Mqtt* mqtt = mqtt::mqttbroker::lib::MqttModel::instance().getMqtt(clientId);

305 mqtt->subscribe(topic, qoS);
306 res->send(jsonString);
307 } else {
308 res->status(404).send("MQTT client has already gone away: " + json["connection_name"].get<std::string>());
309 }
310 },
311 [&res](const std::string& key) {
312 VLOG(1) << "Attribute type not found: " << key;
313
314 res->status(400).send("Attribute type not found: " + key);
315 });
316 });
317
318 router.use(jsonRouter);
319
320 router.get("/clients", [&environment] APPLICATION(req, res) {
321 std::string responseString;
322 int responseStatus = 200;
323
324 try {
325 responseString = getOverviewPage(environment, mqtt::mqttbroker::lib::MqttModel::instance());
326 } catch (const inja::InjaError& error) {
327 responseStatus = 500;
328 responseString = "Internal Server Error\n";
329 responseString +=
330 error.type + " " + error.message + " " + std::to_string(error.location.line) + std::to_string(error.location.column);
331 }
332
333 res->status(responseStatus).send(responseString);
334 });
335
336 router.get("/client", [&environment] APPLICATION(req, res) {
337 std::string responseString;
338 int responseStatus = 200;
339
340 if (req->queries.size() == 1) {
341 const mqtt::mqttbroker::lib::Mqtt* mqtt =
342 mqtt::mqttbroker::lib::MqttModel::instance().getMqtt(urlDecode(req->queries.begin()->first));
343
344 if (mqtt != nullptr) {
345 VLOG(1) << "Subscriptions for client " << mqtt->getClientId();
346 for (const std::string& subscription : mqtt->getSubscriptions()) {
347 VLOG(1) << " " << subscription;
348 }
349
350 try {
351 responseString = getDetailedPage(environment, mqtt);
352 } catch (const inja::InjaError& error) {
353 responseStatus = 500;
354 responseString = "Internal Server Error";
355 responseString += error.type + " " + error.message + " " + std::to_string(error.location.line) +
356 std::to_string(error.location.column);
357 }
358 } else {
359 responseStatus = 404;
360 responseString = "Not Found: " + urlDecode(req->queries.begin()->first);
361 }
362 } else {
363 responseStatus = 400;
364 responseString = "Bad Request: No Client requested";
365 }
366
367 res->status(responseStatus).send(responseString);
368 });
369
370 router.get("/ws", [] APPLICATION(req, res) {
371 if (req->headers.contains("upgrade")) {
372 upgrade(req, res);
373 } else {
374 res->redirect("/clients");
375 }
376 });
377
378 router.get("/", [] APPLICATION(req, res) {
379 if (req->headers.contains("upgrade")) {
380 upgrade(req, res);
381 } else {
382 res->redirect("/clients");
383 }
384 });
385
386 return router;
387}

389static void
390reportState(const std::string& instanceName, const core::socket::SocketAddress& socketAddress, const core::socket::State& state) {
391 switch (state) {
392 case core::socket::State::OK:
393 VLOG(1) << instanceName << ": listening on '" << socketAddress.toString() << "'";
394 break;
395 case core::socket::State::DISABLED:
396 VLOG(1) << instanceName << ": disabled";
397 break;
398 case core::socket::State::ERROR:
399 VLOG(1) << instanceName << ": " << socketAddress.toString() << ": " << state.what();
400 break;
401 case core::socket::State::FATAL:
402 VLOG(1) << instanceName << ": " << socketAddress.toString() << ": " << state.what();
403 break;
404 }
405}

407template <template <typename, typename...> typename SocketServer,
408 typename SocketContextFactory,
409 typename... SocketContextFactoryArgs,
410 typename Server = SocketServer<SocketContextFactory, SocketContextFactoryArgs&&...>,
411 typename SocketAddress = typename Server::SocketAddress,
412 typename = std::enable_if_t<std::is_base_of_v<core::socket::stream::SocketContextFactory, SocketContextFactory>>>
413void startServer(const std::string& instanceName,
414 const std::function<void(typename Server::Config&)>& configurator,
415 SocketContextFactoryArgs&&... socketContextFactoryArgs) {
416 const Server server(instanceName, std::forward<SocketContextFactoryArgs>(socketContextFactoryArgs)...);
417
418 configurator(server.getConfig());
419
420 server.listen([instanceName](const SocketAddress& socketAddress, const core::socket::State& state) {
421 reportState(instanceName, socketAddress, state);
422 });
423}
