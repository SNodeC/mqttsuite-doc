66 iot::mqtt::client::SubProtocol* SubProtocolFactory::create(web::websocket::SubProtocolContext* subProtocolContext) {
67 iot::mqtt::client::SubProtocol* subProtocol = nullptr;
68
69 const mqtt::bridge::lib::Broker& broker =
70 mqtt::bridge::lib::BridgeStore::instance().getBroker(subProtocolContext->getSocketConnection()->getInstanceName());
71
72 VLOG(1) << " Creating Broker instance '" << broker.getInstanceName() << "' of Bridge '" << broker.getBridge().getName() << "'";
73 VLOG(1) << " Bridge client id : " << broker.getClientId();
74 VLOG(1) << " Transport: " << broker.getTransport();
75 VLOG(1) << " Protocol: " << broker.getProtocol();
76 VLOG(1) << " Encryption: " << broker.getEncryption();
77
78 VLOG(1) << " Topics:";
79 const std::list<iot::mqtt::Topic>& topics = broker.getTopics();
80 for (const iot::mqtt::Topic& topic : topics) {
81 VLOG(1) << " " << static_cast<uint16_t>(topic.getQoS()) << ":" << topic.getName();
82 }
83
84 if (!topics.empty()) {
85 subProtocol = new iot::mqtt::client::SubProtocol(
86 subProtocolContext,
87 getName(),
88 new mqtt::bridge::lib::Mqtt(subProtocolContext->getSocketConnection()->getConnectionName(), broker));
89 }
90
91 return subProtocol;
92 }
