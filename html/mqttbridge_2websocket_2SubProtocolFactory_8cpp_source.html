45 iot::mqtt::client::SubProtocol* SubProtocolFactory::create(web::websocket::SubProtocolContext* subProtocolContext) {
46 iot::mqtt::client::SubProtocol* subProtocol = nullptr;
47
48 const std::string& instanceName = subProtocolContext->getSocketConnection()->getInstanceName();
49
50 const mqtt::bridge::lib::Broker& broker = mqtt::bridge::lib::BridgeStore::instance().getBroker(instanceName);
51
52 VLOG(1) << " Creating Broker instance '" << broker.getInstanceName() << "' of Bridge '" << broker.getBridge().getName() << "'";
53 VLOG(1) << " Bridge client id : " << broker.getClientId();
54 VLOG(1) << " Transport: " << broker.getTransport();
55 VLOG(1) << " Protocol: " << broker.getProtocol();
56 VLOG(1) << " Encryption: " << broker.getEncryption();
57
58 VLOG(1) << " Topics:";
59 const std::list<iot::mqtt::Topic>& topics = broker.getTopics();
60 for (const iot::mqtt::Topic& topic : topics) {
61 VLOG(1) << " " << static_cast<uint16_t>(topic.getQoS()) << ":" << topic.getName();
62 }
63
64 if (!topics.empty()) {
65 subProtocol = new iot::mqtt::client::SubProtocol(
66 subProtocolContext,
67 getName(),
68 new mqtt::bridge::lib::Mqtt(subProtocolContext->getSocketConnection()->getConnectionName(), broker));
69 }
70
71 return subProtocol;
72 }
