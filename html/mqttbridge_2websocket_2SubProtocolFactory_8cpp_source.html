67 iot::mqtt::client::SubProtocol* SubProtocolFactory::create(web::websocket::SubProtocolContext* subProtocolContext) {
68 iot::mqtt::client::SubProtocol* subProtocol = nullptr;
69
70 const std::string& instanceName = subProtocolContext->getSocketConnection()->getInstanceName();
71
72 const mqtt::bridge::lib::Broker& broker = mqtt::bridge::lib::BridgeStore::instance().getBroker(instanceName);
73
74 VLOG(1) << " Creating Broker instance '" << broker.getInstanceName() << "' of Bridge '" << broker.getBridge().getName() << "'";
75 VLOG(1) << " Bridge client id : " << broker.getClientId();
76 VLOG(1) << " Transport: " << broker.getTransport();
77 VLOG(1) << " Protocol: " << broker.getProtocol();
78 VLOG(1) << " Encryption: " << broker.getEncryption();
79
80 VLOG(1) << " Topics:";
81 const std::list<iot::mqtt::Topic>& topics = broker.getTopics();
82 for (const iot::mqtt::Topic& topic : topics) {
83 VLOG(1) << " " << static_cast<uint16_t>(topic.getQoS()) << ":" << topic.getName();
84 }
85
86 if (!topics.empty()) {
87 subProtocol = new iot::mqtt::client::SubProtocol(
88 subProtocolContext,
89 getName(),
90 new mqtt::bridge::lib::Mqtt(subProtocolContext->getSocketConnection()->getConnectionName(), broker));
91 }
92
93 return subProtocol;
94 }

103
104 const bool success = mqtt::bridge::lib::BridgeStore::instance().loadAndValidate(bridgeConfigFile);
