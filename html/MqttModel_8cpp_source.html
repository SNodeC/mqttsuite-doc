84 static std::string windowId(const std::string& clientId) {
85 std::ostringstream windowId("window");
86 for (char ch : clientId) {
87 if (std::isalnum(static_cast<unsigned char>(ch))) {
88 windowId << ch;
89 } else {
90 windowId << std::hex << std::uppercase << std::setw(2) << std::setfill('0')
91 << static_cast<int>(static_cast<unsigned char>(ch));
92 }
93 }
94
95 return windowId.str();
96 }

118 void MqttModel::addClient(const std::string& clientId, Mqtt* mqtt) {
119 MqttModelEntry mqttModelEntry(mqtt);
120
121 nlohmann::json json({href(mqtt->getClientId(), "/client?" + mqtt->getClientId(), windowId(mqtt->getClientId()), 450, 900),
122 mqttModelEntry.onlineSince(),
123 "<duration>" + mqttModelEntry.onlineDuration() + "</duration>",
124 mqtt->getConnectionName(),
125 mqtt->getMqttContext()->getSocketConnection()->getLocalAddress().toString(),
126 mqtt->getMqttContext()->getSocketConnection()->getRemoteAddress().toString(),
127 "<button class=\"red-btn\" onClick=\"disconnectClient('" + mqtt->getClientId() + "')\">Disconnect</button>"});
128
129 sendEvent(json.dump(), "connect", std::to_string(id++));
130
131 modelMap.emplace(clientId, std::move(mqttModelEntry));
132 }

207 void MqttModel::sendEvent(const std::string& data, const std::string& event, const std::string& id) {
208 for (auto& eventReceiver : eventReceiverList) {
209 if (const auto& response = eventReceiver.response.lock()) {
210 if (response->isConnected()) {
211 if (!event.empty()) {
212 response->sendFragment("event:" + event);
213 }
214 if (!id.empty()) {
215 response->sendFragment("id:" + id);
216 }
217 response->sendFragment("data:" + data);
218 response->sendFragment();
219 }
220 }
221 }
222 }

224 std::string MqttModel::timePointToString(const std::chrono::time_point<std::chrono::system_clock>& timePoint) {
225 std::time_t time = std::chrono::system_clock::to_time_t(timePoint);
226 std::tm* tm_ptr = std::gmtime(&time);
227
228 char buffer[100];
229 std::string onlineSince = "Formatting error";
230
231 // Format: "2025-02-02 14:30:00"
232 if (std::strftime(buffer, sizeof(buffer), "%Y-%m-%d %H:%M:%S", tm_ptr)) {
233 onlineSince = std::string(buffer) + " UTC";
234 }
235
236 return onlineSince;
237 }

239 std::string MqttModel::durationToString(const std::chrono::time_point<std::chrono::system_clock>& bevore,
240 const std::chrono::time_point<std::chrono::system_clock>& later) {
241 using seconds_duration_type = std::chrono::duration<std::chrono::seconds::rep>::rep;
242
243 seconds_duration_type totalSeconds = std::chrono::duration_cast<std::chrono::seconds>(later - bevore).count();
244
245 // Compute days, hours, minutes, and seconds
246 seconds_duration_type days = totalSeconds / 86400; // 86400 seconds in a day
247 seconds_duration_type remainder = totalSeconds % 86400;
248 seconds_duration_type hours = remainder / 3600;
249 remainder = remainder % 3600;
250 seconds_duration_type minutes = remainder / 60;
251 seconds_duration_type seconds = remainder % 60;
252
253 // Format the components into a string using stringstream
254 std::ostringstream oss;
255 if (days > 0) {
256 oss << days << " day" << (days == 1 ? "" : "s") << ", ";
257 }
258 oss << std::setw(2) << std::setfill('0') << hours << ":" << std::setw(2) << std::setfill('0') << minutes << ":" << std::setw(2)
259 << std::setfill('0') << seconds;
260
261 return oss.str();
262 }
