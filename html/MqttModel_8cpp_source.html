113 std::string MqttModel::timePointToString(const std::chrono::time_point<std::chrono::system_clock>& timePoint) {
114 std::time_t time = std::chrono::system_clock::to_time_t(timePoint);
115 std::tm* tm_ptr = std::gmtime(&time);
116
117 char buffer[100];
118 std::string onlineSince = "Formatting error";
119
120 // Format: "2025-02-02 14:30:00"
121 if (std::strftime(buffer, sizeof(buffer), "%Y-%m-%d %H:%M:%S", tm_ptr)) {
122 onlineSince = std::string(buffer) + " UTC";
123 }
124
125 return onlineSince;
126 }

128 std::string MqttModel::durationToString(const std::chrono::time_point<std::chrono::system_clock>& bevore,
129 const std::chrono::time_point<std::chrono::system_clock>& later) {
130 using seconds_duration_type = std::chrono::duration<std::chrono::seconds::rep>::rep;
131
132 seconds_duration_type totalSeconds = std::chrono::duration_cast<std::chrono::seconds>(later - bevore).count();
133
134 // Compute days, hours, minutes, and seconds
135 seconds_duration_type days = totalSeconds / 86400; // 86400 seconds in a day
136 seconds_duration_type remainder = totalSeconds % 86400;
137 seconds_duration_type hours = remainder / 3600;
138 remainder = remainder % 3600;
139 seconds_duration_type minutes = remainder / 60;
140 seconds_duration_type seconds = remainder % 60;
141
142 // Format the components into a string using stringstream
143 std::ostringstream oss;
144 if (days > 0) {
145 oss << days << " day" << (days == 1 ? "" : "s") << ", ";
146 }
147 oss << std::setw(2) << std::setfill('0') << hours << ":" << std::setw(2) << std::setfill('0') << minutes << ":" << std::setw(2)
148 << std::setfill('0') << seconds;
149
150 return oss.str();
151 }
