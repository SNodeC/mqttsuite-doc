91 std::string MqttModel::timePointToString(const std::chrono::time_point<std::chrono::system_clock>& timePoint) {
92 std::time_t time = std::chrono::system_clock::to_time_t(timePoint);
93 std::tm* tm_ptr = std::gmtime(&time);
94
95 char buffer[100];
96 std::string onlineSince = "Formatting error";
97
98 // Format: "2025-02-02 14:30:00"
99 if (std::strftime(buffer, sizeof(buffer), "%Y-%m-%d %H:%M:%S", tm_ptr)) {
100 onlineSince = std::string(buffer) + " UTC";
101 }
102
103 return onlineSince;
104 }

106 std::string MqttModel::durationToString(const std::chrono::time_point<std::chrono::system_clock>& bevore,
107 const std::chrono::time_point<std::chrono::system_clock>& later) {
108 using seconds_duration_type = std::chrono::duration<std::chrono::seconds::rep>::rep;
109
110 seconds_duration_type totalSeconds = std::chrono::duration_cast<std::chrono::seconds>(later - bevore).count();
111
112 // Compute days, hours, minutes, and seconds
113 seconds_duration_type days = totalSeconds / 86400; // 86400 seconds in a day
114 seconds_duration_type remainder = totalSeconds % 86400;
115 seconds_duration_type hours = remainder / 3600;
116 remainder = remainder % 3600;
117 seconds_duration_type minutes = remainder / 60;
118 seconds_duration_type seconds = remainder % 60;
119
120 // Format the components into a string using stringstream
121 std::ostringstream oss;
122 if (days > 0) {
123 oss << days << " day" << (days == 1 ? "" : "s") << ", ";
124 }
125 oss << std::setw(2) << std::setfill('0') << hours << ":" << std::setw(2) << std::setfill('0') << minutes << ":" << std::setw(2)
126 << std::setfill('0') << seconds;
127
128 return oss.str();
129 }
