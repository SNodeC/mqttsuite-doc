77 nlohmann::json& JsonMappingReader::readMappingFromFile(const std::string& mapFilePath) {
78 if (mapFileJson.empty()) {
79 if (!mapFilePath.empty()) {
80 std::ifstream mapFile(mapFilePath);
81
82 if (mapFile.is_open()) {
83 VLOG(1) << "MappingFilePath: " << mapFilePath;
84
85 try {
86 mapFile >> mapFileJson;
87
88 try {
89 const nlohmann::json_schema::json_validator validator(
90 mappingJsonSchema, nullptr, nlohmann::json_schema::default_string_format_check);
91
92 try {
93 const nlohmann::json defaultPatch = validator.validate(mapFileJson);
94
95 try {
96 mapFileJsonDefaultPatched = mapFileJson.patch(defaultPatch);
97 VLOG(0) << "Patched: \n" << mapFileJson.dump(4);
98 } catch (const std::exception& e) {
99 VLOG(1) << e.what();
100 VLOG(1) << "Patching JSON with default patch failed:\n" << defaultPatch.dump(4);
101 mapFileJson.clear();
102 }
103 } catch (const std::exception& e) {
104 VLOG(1) << " Validating JSON failed:\n" << mapFileJson.dump(4);
105 VLOG(1) << " " << e.what();
106 mapFileJson.clear();
107 }
108 } catch (const std::exception& e) {
109 VLOG(1) << e.what();
110 VLOG(1) << "Setting root json mapping schema failed:\n" << mappingJsonSchema.dump(4);
111 mapFileJson.clear();
112 }
113 } catch (const std::exception& e) {
114 VLOG(1) << "JSON map file parsing failed: " << e.what() << " at " << mapFile.tellg();
115 mapFileJson.clear();
116 }
117
118 mapFile.close();
119 } else {
120 VLOG(1) << "MappingFilePath: " << mapFilePath << " not found";
121 }
122 } else {
123 VLOG(1) << "MappingFilePath empty";
124 }
125 }
126
127 return mapFileJsonDefaultPatched;
128 }

165 void JsonMappingReader::deployDraft(const std::string& mapFilePath) {
166 std::string draftPath = getDraftPath(mapFilePath);
167 if (!fs::exists(draftPath))
168 return;
169
170 // 1. Inject creation timestamp into draft
171 try {
172 std::ifstream f(draftPath);
173 nlohmann::json j;
174 f >> j;
175 f.close();
176
177 auto now = std::chrono::system_clock::now();
178 std::time_t now_c = std::chrono::system_clock::to_time_t(now);
179 std::stringstream ss;
180 ss << std::put_time(std::gmtime(&now_c), "%Y-%m-%dT%H:%M:%SZ");
181
182 if (!j.contains("meta"))
183 j["meta"] = nlohmann::json::object();
184 j["meta"]["created"] = ss.str();
185 j["meta"]["version"] = std::to_string(std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch()).count());
186
187 std::ofstream out(draftPath, std::ios::trunc);
188 out << j.dump(2);
189 out.close();
190 } catch (const std::exception& e) {
191 VLOG(1) << "Failed to inject metadata into draft: " << e.what();
192 }
193
194 // 2. Backup current active file
195 if (fs::exists(mapFilePath)) {
196 fs::path versionDir = fs::path(mapFilePath).parent_path() / "versions";
197 if (!fs::exists(versionDir)) {
198 fs::create_directories(versionDir);
199 }
200
201 auto now = std::chrono::system_clock::now();
202 auto timestamp = std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch()).count();
203 std::string filename = fs::path(mapFilePath).filename().string();
204 std::string backupPath = versionDir / (filename + "." + std::to_string(timestamp));
205
206 fs::copy_file(mapFilePath, backupPath, fs::copy_options::overwrite_existing);
207
208 // 3. Prune old versions (Keep last 50)
209 try {
210 std::vector<fs::path> versions;
211 for (const auto& entry : fs::directory_iterator(versionDir)) {
212 if (entry.path().filename().string().starts_with(filename + ".")) {
213 versions.push_back(entry.path());
214 }
215 }
216 if (versions.size() > 50) {
217 std::sort(versions.begin(), versions.end(), [](const fs::path& a, const fs::path& b) {
218 return fs::last_write_time(a) < fs::last_write_time(b); // Oldest first
219 });
220 for (size_t i = 0; i < versions.size() - 50; ++i) {
221 fs::remove(versions[i]);
222 }
223 }
224 } catch (...) {
225 }
226 }
227
228 // 4. Promote draft to active
229 fs::rename(draftPath, mapFilePath);
230
231 // Invalidate cache so next readMappingFromFile reloads it
232 mapFileJson.clear();
233 }

242 std::vector<JsonMappingReader::VersionEntry> JsonMappingReader::getHistory(const std::string& mapFilePath) {
243 std::vector<VersionEntry> history;
244 fs::path versionDir = fs::path(mapFilePath).parent_path() / "versions";
245 std::string baseName = fs::path(mapFilePath).filename().string();
246
247 if (!fs::exists(versionDir))
248 return history;
249
250 for (const auto& entry : fs::directory_iterator(versionDir)) {
251 if (entry.path().filename().string().starts_with(baseName + ".")) {
252 VersionEntry v;
253 v.filename = entry.path().string();
254 // Extract ID (timestamp) from filename extension
255 v.id = entry.path().extension().string().substr(1);
256
257 // Peek inside JSON to get the comment
258 try {
259 std::ifstream f(v.filename);
260 nlohmann::json j;
261 f >> j;
262 if (j.contains("meta")) {
263 if (j["meta"].contains("comment"))
264 v.comment = j["meta"]["comment"];
265 if (j["meta"].contains("created"))
266 v.date = j["meta"]["created"];
267 }
268 } catch (...) {
269 }
270
271 // Fallback date if not in meta
272 if (v.date.empty()) {
273 try {
274 long long ts = std::stoll(v.id);
275 std::time_t t = static_cast<std::time_t>(ts);
276 std::stringstream ss;
277 ss << std::put_time(std::gmtime(&t), "%Y-%m-%dT%H:%M:%SZ");
278 v.date = ss.str();
279 } catch (...) {
280 v.date = "Unknown";
281 }
282 }
283
284 history.push_back(v);
285 }
286 }
287 // Sort by ID (descending)
288 std::sort(history.begin(), history.end(), [](const VersionEntry& a, const VersionEntry& b) {
289 // String comparison of timestamps works if they are same length, but better to be safe
290 try {
291 return std::stoll(a.id) > std::stoll(b.id);
292 } catch (...) {
293 return a.id > b.id;
294 }
295 });
296 return history;
297 }

299 void JsonMappingReader::rollbackTo(const std::string& mapFilePath, const std::string& versionId) {
300 fs::path versionDir = fs::path(mapFilePath).parent_path() / "versions";
301 std::string baseName = fs::path(mapFilePath).filename().string();
302 fs::path backupPath = versionDir / (baseName + "." + versionId);
303
304 if (!fs::exists(backupPath)) {
305 throw std::runtime_error("Version not found: " + versionId);
306 }
307
308 // Validate before rollback
309 try {
310 std::ifstream f(backupPath);
311 nlohmann::json j;
312 f >> j;
313 const nlohmann::json_schema::json_validator validator(
314 mappingJsonSchema, nullptr, nlohmann::json_schema::default_string_format_check);
315 validator.validate(j);
316 } catch (const std::exception& e) {
317 throw std::runtime_error(std::string("Cannot rollback: Version is invalid against current schema: ") + e.what());
318 }
319
320 // Overwrite active file
321 fs::copy_file(backupPath, mapFilePath, fs::copy_options::overwrite_existing);
322
323 // Delete any existing draft to avoid confusion
324 discardDraft(mapFilePath);
325
326 mapFileJson.clear();
327 }
