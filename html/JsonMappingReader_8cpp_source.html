41 nlohmann::json& JsonMappingReader::readMappingFromFile(const std::string& mapFilePath) {
42 if (!mapFileJsons.contains(mapFilePath)) {
43 if (!mapFilePath.empty()) {
44 std::ifstream mapFile(mapFilePath);
45
46 if (mapFile.is_open()) {
47 VLOG(1) << "MappingFilePath: " << mapFilePath;
48
49 try {
50 mapFile >> mapFileJsons[mapFilePath];
51
52 try {
53 const nlohmann::json_schema::json_validator validator(mappingJsonSchema);
54
55 try {
56 const nlohmann::json defaultPatch = validator.validate(mapFileJsons[mapFilePath]);
57
58 if (!defaultPatch.empty()) {
59 try {
60 mapFileJsons[mapFilePath] = mapFileJsons[mapFilePath].patch(defaultPatch);
61 } catch (const std::exception& e) {
62 VLOG(1) << e.what();
63 VLOG(1) << "Patching JSON with default patch failed:\n" << defaultPatch.dump(4);
64 mapFileJsons[mapFilePath].clear();
65 }
66 }
67 } catch (const std::exception& e) {
68 VLOG(1) << " Validating JSON failed:\n" << mapFileJsons[mapFilePath].dump(4);
69 VLOG(1) << " " << e.what();
70 mapFileJsons[mapFilePath].clear();
71 }
72 } catch (const std::exception& e) {
73 VLOG(1) << e.what();
74 VLOG(1) << "Setting root json mapping schema failed:\n" << mappingJsonSchema.dump(4);
75 mapFileJsons[mapFilePath].clear();
76 }
77 } catch (const std::exception& e) {
78 VLOG(1) << "JSON map file parsing failed: " << e.what() << " at " << mapFile.tellg();
79 mapFileJsons[mapFilePath].clear();
80 }
81 mapFile.close();
82 } else {
83 VLOG(1) << "MappingFilePath: " << mapFilePath << " not found";
84 }
85 } else {
86 VLOG(1) << "MappingFilePath empty";
87 }
88 }
89
90 return mapFileJsons[mapFilePath];
91 }
