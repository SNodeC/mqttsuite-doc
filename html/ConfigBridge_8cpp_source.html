52 mqtt::bridge::ConfigBridge::ConfigBridge()
53 : bridgeSc(
54 utils::Config::newInstance(net::config::Instance(std::string(name), std::string(description), this), "Applications", true)) {
55 bridgeDefinitionOpt = bridgeSc->add_option("--definition", "MQTT bridge definition file (JSON format)")
56 ->check(CLI::ExistingFile)
57 ->group(bridgeSc->get_formatter()->get_label("Persistent Options"))
58 ->type_name("path")
59 ->configurable()
60 ->required();
61
62 htmlDirOpt = bridgeSc->add_option("--html-dir", "Path to html source directory")
63 ->check(CLI::ExistingDirectory)
64 ->default_str(std::string(CMAKE_INSTALL_PREFIX) + "/var/www/mqttsuite/mqttbridge")
65 ->group(bridgeSc->get_formatter()->get_label("Persistent Options"))
66 ->type_name("path")
67 ->configurable();
68
69 bridgeSc->needs(bridgeDefinitionOpt)->required();
70 bridgeSc->get_parent()->needs(bridgeSc);
71 }

73 void mqtt::bridge::ConfigBridge::setDefinitionFile(const std::string& definitionFile) const {
