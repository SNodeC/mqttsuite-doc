30/** @see date_time_check */
31void rfc3339_date_check(const std::string &value)
32{
33 const static std::regex dateRegex{R"(^([0-9]{4})\-([0-9]{2})\-([0-9]{2})$)"};
34
35 std::smatch matches;
36 if (!std::regex_match(value, matches, dateRegex)) {
37 throw std::invalid_argument(value + " is not a date string according to RFC 3339.");
38 }
39
40 const auto year = std::stoi(matches[1].str());
41 const auto month = std::stoi(matches[2].str());
42 const auto mday = std::stoi(matches[3].str());
43
44 const auto isLeapYear = (year % 4 == 0) && ((year % 100 != 0) || (year % 400 == 0));
45
46 range_check(month, 1, 12);
47 if (month == 2) {
48 range_check(mday, 1, isLeapYear ? 29 : 28);
49 } else if (month <= 7) {
50 range_check(mday, 1, month % 2 == 0 ? 30 : 31);
51 } else {
52 range_check(mday, 1, month % 2 == 0 ? 31 : 30);
53 }
54}

56/** @see date_time_check */
57void rfc3339_time_check(const std::string &value)
58{
59 const static std::regex timeRegex{R"(^([0-9]{2})\:([0-9]{2})\:([0-9]{2})(\.[0-9]+)?(?:[Zz]|((?:\+|\-)[0-9]{2})\:([0-9]{2}))$)"};
60
61 std::smatch matches;
62 if (!std::regex_match(value, matches, timeRegex)) {
63 throw std::invalid_argument(value + " is not a time string according to RFC 3339.");
64 }
65
66 auto hour = std::stoi(matches[1].str());
67 auto minute = std::stoi(matches[2].str());
68 auto second = std::stoi(matches[3].str());
69 // const auto secfrac = std::stof( matches[4].str() );
70
71 range_check(hour, 0, 23);
72 range_check(minute, 0, 59);
73
74 int offsetHour = 0,
75 offsetMinute = 0;
76
77 /* don't check the numerical offset if time zone is specified as 'Z' */
78 if (!matches[5].str().empty()) {
79 offsetHour = std::stoi(matches[5].str());
80 offsetMinute = std::stoi(matches[6].str());
81
82 range_check(offsetHour, -23, 23);
83 range_check(offsetMinute, 0, 59);
84 if (offsetHour < 0)
85 offsetMinute *= -1;
86 }
87
88 /**
89 * @todo Could be made more exact by querying a leap second database and choosing the
90 * correct maximum in {58,59,60}. This current solution might match some invalid dates
91 * but it won't lead to false negatives. This only works if we know the full date, however
92 */
93
94 auto day_minutes = hour * 60 + minute - (offsetHour * 60 + offsetMinute);
95 if (day_minutes < 0)
96 day_minutes += 60 * 24;
97 hour = day_minutes % 24;
98 minute = day_minutes / 24;
99
100 if (hour == 23 && minute == 59)
101 range_check(second, 0, 60); // possible leap-second
102 else
103 range_check(second, 0, 59);
104}

106/**
107 * @see https://tools.ietf.org/html/rfc3339#section-5.6
108 *
109 * @verbatim
110 * date-fullyear = 4DIGIT
111 * date-month = 2DIGIT ; 01-12
112 * date-mday = 2DIGIT ; 01-28, 01-29, 01-30, 01-31 based on
113 * ; month/year
114 * time-hour = 2DIGIT ; 00-23
115 * time-minute = 2DIGIT ; 00-59
116 * time-second = 2DIGIT ; 00-58, 00-59, 00-60 based on leap second
117 * ; rules
118 * time-secfrac = "." 1*DIGIT
119 * time-numoffset = ("+" / "-") time-hour ":" time-minute
120 * time-offset = "Z" / time-numoffset
121 *
122 * partial-time = time-hour ":" time-minute ":" time-second
123 * [time-secfrac]
124 * full-date = date-fullyear "-" date-month "-" date-mday
125 * full-time = partial-time time-offset
126 *
127 * date-time = full-date "T" full-time
128 * @endverbatim
129 * NOTE: Per [ABNF] and ISO8601, the "T" and "Z" characters in this
130 * syntax may alternatively be lower case "t" or "z" respectively.
131 */
132void rfc3339_date_time_check(const std::string &value)
133{
134 const static std::regex dateTimeRegex{R"(^([0-9]{4}\-[0-9]{2}\-[0-9]{2})[Tt]([0-9]{2}\:[0-9]{2}\:[0-9]{2}(?:\.[0-9]+)?(?:[Zz]|(?:\+|\-)[0-9]{2}\:[0-9]{2}))$)"};
135
136 std::smatch matches;
137 if (!std::regex_match(value, matches, dateTimeRegex)) {
138 throw std::invalid_argument(value + " is not a date-time string according to RFC 3339.");
139 }
140
141 rfc3339_date_check(matches[1].str());
142 rfc3339_time_check(matches[2].str());
143}

336/**
337 * Checks validity for built-ins by converting the definitions given as ABNF in the linked RFC from
338 * @see https://json-schema.org/understanding-json-schema/reference/string.html#built-in-formats
339 * into regular expressions using @see https://www.msweet.org/abnf/ and some manual editing.
340 *
341 * @see https://json-schema.org/latest/json-schema-validation.html
342 */
343void default_string_format_check(const std::string &format, const std::string &value)
344{
345 if (format == "date-time") {
346 rfc3339_date_time_check(value);
347 } else if (format == "date") {
348 rfc3339_date_check(value);
349 } else if (format == "time") {
350 rfc3339_time_check(value);
351 } else if (format == "uri") {
352 rfc3986_uri_check(value);
353 } else if (format == "email") {
354 if (!is_ascii(value)) {
355 throw std::invalid_argument(value + " contains non-ASCII values, not RFC 5321 compliant.");
356 }
357 if (!is_address(&*value.begin(), &*value.end())) {
358 throw std::invalid_argument(value + " is not a valid email according to RFC 5321.");
359 }
360 } else if (format == "idn-email") {
361 if (!is_address(&*value.begin(), &*value.end())) {
362 throw std::invalid_argument(value + " is not a valid idn-email according to RFC 6531.");
363 }
364 } else if (format == "hostname") {
365 static const std::regex hostRegex{hostname};
366 if (!std::regex_match(value, hostRegex)) {
367 throw std::invalid_argument(value + " is not a valid hostname according to RFC 3986 Appendix A.");
368 }
369 } else if (format == "ipv4") {
370 const static std::regex ipv4Regex{"^" + ipv4Address + "$"};
371 if (!std::regex_match(value, ipv4Regex)) {
372 throw std::invalid_argument(value + " is not an IPv4 string according to RFC 2673.");
373 }
374 } else if (format == "ipv6") {
375 static const std::regex ipv6Regex{ipv6Address};
376 if (!std::regex_match(value, ipv6Regex)) {
377 throw std::invalid_argument(value + " is not an IPv6 string according to RFC 5954.");
378 }
379 } else if (format == "uuid") {
380 static const std::regex uuidRegex{uuid};
381 if (!std::regex_match(value, uuidRegex)) {
382 throw std::invalid_argument(value + " is not an uuid string according to RFC 4122.");
383 }
384 } else if (format == "regex") {
385 try {
386 std::regex re(value, std::regex::ECMAScript);
387 } catch (std::exception &exception) {
388 throw exception;
389 }
390 } else {
391 /* yet unsupported JSON schema draft 7 built-ins */
392 static const std::vector<std::string> jsonSchemaStringFormatBuiltIns{
393 "date-time", "time", "date", "email", "idn-email", "hostname", "idn-hostname", "ipv4", "ipv6", "uri",
394 "uri-reference", "iri", "iri-reference", "uri-template", "json-pointer", "relative-json-pointer", "regex"};
395 if (std::find(jsonSchemaStringFormatBuiltIns.begin(), jsonSchemaStringFormatBuiltIns.end(), format) != jsonSchemaStringFormatBuiltIns.end()) {
396 throw std::logic_error("JSON schema string format built-in " + format + " not yet supported. " +
397 "Please open an issue or use a custom format checker.");
398 }
399
400 throw std::logic_error("Don't know how to validate " + format);
401 }
