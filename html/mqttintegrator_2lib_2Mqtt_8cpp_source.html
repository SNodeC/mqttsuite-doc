101 void Mqtt::onConnack(const iot::mqtt::packets::Connack& connack) {
102 if (connack.getReturnCode() == 0 && !connack.getSessionPresent()) {
103 sendPublish("snode.c/_cfg_/connection", connectionJson.dump(), 0, true);
104
105 const std::list<iot::mqtt::Topic> topicList = MqttMapper::extractSubscriptions();
106
107 for (const iot::mqtt::Topic& topic : topicList) {
108 VLOG(1) << "MQTT: Subscribe Topic: " << topic.getName() << ", qoS: " << static_cast<uint16_t>(topic.getQoS());
109 }
110
111 sendSubscribe(topicList);
112 }
113 }
