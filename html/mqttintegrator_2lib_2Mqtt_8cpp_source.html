98 void Mqtt::onConnack(const iot::mqtt::packets::Connack& connack) {
99 if (connack.getReturnCode() == 0 && !connack.getSessionPresent()) {
100 sendPublish("snode.c/_cfg_/connection", connectionJson.dump(), 0, true);
101
102 const std::list<iot::mqtt::Topic> topicList = MqttMapper::extractSubscriptions();
103
104 for (const iot::mqtt::Topic& topic : topicList) {
105 VLOG(1) << "MQTT: Subscribe Topic: " << topic.getName() << ", qoS: " << static_cast<uint16_t>(topic.getQoS());
106 }
107
108 sendSubscribe(topicList);
109 }
110 }
