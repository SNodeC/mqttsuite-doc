112 void Mqtt::onConnack(const iot::mqtt::packets::Connack& connack) {
113 if (connack.getReturnCode() == 0 && !connack.getSessionPresent()) {
114 sendPublish("snode.c/_cfg_/connection", connectionJson.dump(), 0, true);
115
116 const std::list<iot::mqtt::Topic> topicList = MqttMapper::extractSubscriptions();
117
118 for (const iot::mqtt::Topic& topic : topicList) {
119 VLOG(1) << "MQTT: Subscribe Topic: " << topic.getName() << ", qoS: " << static_cast<uint16_t>(topic.getQoS());
120 }
121
122 sendSubscribe(topicList);
123 }
