76 void Mqtt::onConnack(const iot::mqtt::packets::Connack& connack) {
77 if (connack.getReturnCode() == 0 && !connack.getSessionPresent()) {
78 sendPublish("snode.c/_cfg_/connection", connectionJson.dump(), 0, true);
79
80 const std::list<iot::mqtt::Topic> topicList = MqttMapper::extractSubscriptions();
81
82 for (const iot::mqtt::Topic& topic : topicList) {
83 VLOG(1) << "MQTT: Subscribe Topic: " << topic.getName() << ", qoS: " << static_cast<uint16_t>(topic.getQoS());
84 }
85
86 sendSubscribe(topicList);
87 }
88 }
