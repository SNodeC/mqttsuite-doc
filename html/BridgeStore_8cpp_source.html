69 bool BridgeStore::loadAndValidate(const std::string& fileName) {
70 bool success = !brokers.empty();
71
72#include "bridge-schema.json.h" // IWYU pragma: keep
73
74 if (!success) {
75 try {
76 const nlohmann::json bridgeJsonSchema = nlohmann::json::parse(bridgeJsonSchemaString);
77
78 if (!fileName.empty()) {
79 std::ifstream bridgeConfigJsonFile(fileName);
80
81 if (bridgeConfigJsonFile.is_open()) {
82 VLOG(1) << "Bridge config JSON: " << fileName;
83
84 try {
85 nlohmann::json bridgesConfigJson;
86
87 bridgeConfigJsonFile >> bridgesConfigJson;
88
89 try {
90 const nlohmann::json_schema::json_validator validator(bridgeJsonSchema);
91
92 try {
93 const nlohmann::json defaultPatch = validator.validate(bridgesConfigJson);
94
95 if (!defaultPatch.empty()) {
96 try {
97 bridgesConfigJson = bridgesConfigJson.patch(defaultPatch);
98
99 for (const nlohmann::json& bridgeConfigJson : bridgesConfigJson["bridges"]) {
100 Bridge& bridge = bridgeList.emplace_back(bridgeConfigJson["name"]);
101
102 for (const nlohmann::json& brokerConfigJson : bridgeConfigJson["brokers"]) {
103 std::list<iot::mqtt::Topic> topics;
104 for (const nlohmann::json& topicJson : brokerConfigJson["topics"]) {
105 topics.emplace_back(topicJson["topic"], // cppcheck-suppress useStlAlgorithm
106 topicJson["qos"]);
107 }
108
109 const nlohmann::json& connection = brokerConfigJson["connection"];
110 brokers.emplace(brokerConfigJson["instance_name"],
111 Broker(bridge,
112 connection["client_id"],
113 connection["keep_alive"],
114 connection["clean_session"],
115 connection["will_topic"],
116 connection["will_message"],
117 connection["will_qos"],
118 connection["will_retain"],
119 connection["username"],
120 connection["password"],
121 connection["loop_prevention"],
122 brokerConfigJson["instance_name"],
123 brokerConfigJson["protocol"],
124 brokerConfigJson["encryption"],
125 brokerConfigJson["transport"],
126 std::move(topics)));
127 }
128 }
129
130 success = true;
131 } catch (const std::exception& e) {
132 VLOG(1) << " Patching JSON with default patch failed:\n" << defaultPatch.dump(4);
133 VLOG(1) << " " << e.what();
134 }
135 }
136 } catch (const std::exception& e) {
137 VLOG(1) << " Validating JSON failed:\n" << bridgesConfigJson.dump(4);
138 VLOG(1) << " " << e.what();
139 }
140 } catch (const std::exception& e) {
141 VLOG(1) << " Setting root json mapping schema failed:\n" << bridgeJsonSchema.dump(4);
142 VLOG(1) << " " << e.what();
143 }
144 } catch (const std::exception& e) {
145 VLOG(1) << " JSON map file parsing failed:" << e.what() << " at " << bridgeConfigJsonFile.tellg();
146 }
147
148 bridgeConfigJsonFile.close();
149 } else {
150 VLOG(1) << "BridgeJsonConfig: " << fileName << " not found";
151 }
152 } else {
153 // Do not log missing path. In regular use this missing option is captured by the command line interface
154 }
155 } catch (const std::exception& e) {
156 VLOG(1) << "Parsing schema failed: " << e.what();
157 VLOG(1) << bridgeJsonSchemaString;
158 }
159 } else {
160 VLOG(1) << "MappingFile already loaded and validated";
161 }
162
163 return success;
164 }
