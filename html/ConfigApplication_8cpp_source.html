56 static CLI::Validator parent_dir_exists_and_target_not_dir() {
57 return CLI::Validator(
58 [](const std::string& value) -> std::string {
59 namespace fs = std::filesystem;
60
61 fs::path p{value};
62
63 fs::path parent = p.parent_path();
64 if (parent.empty())
65 parent = fs::path{"."};
66
67 std::error_code ec;
68
69 if (!fs::exists(parent, ec) || ec)
70 return "Parent directory does not exist: " + parent.string();
71
72 if (!fs::is_directory(parent, ec) || ec)
73 return "Parent path is not a directory: " + parent.string();
74
75 // Only forbid: target is an existing directory (including symlink-to-dir)
76 if (fs::exists(p, ec) && !ec && fs::is_directory(p, ec) && !ec)
77 return "Target must not be an existing directory: " + p.string();
78
79 return {}; // OK
80 },
81 "PATH(parent exists, target not directory)",
82 "ParentExistsTargetNotDir");
83 }

85 ConfigApplication::ConfigApplication(CLI::App* configSc)
86 : configSc(configSc) {
87 mappingFileOpt = configSc->add_option("--mqtt-mapping-file", "MQTT mapping file (json format) for integration")
88 ->check(CLI::ExistingFile)
89 ->group(configSc->get_formatter()->get_label("Persistent Options"))
90 ->type_name("path")
91 ->configurable();
92
93 sessionStoreOpt = configSc->add_option("--mqtt-session-store", "Path to file for the persistent session store")
94 ->check(parent_dir_exists_and_target_not_dir())
95 ->group(configSc->get_formatter()->get_label("Persistent Options"))
96 ->type_name("path")
97 ->configurable();
98 }
