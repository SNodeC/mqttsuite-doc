60 core::socket::stream::SocketContext* SocketContextFactory::create(core::socket::stream::SocketConnection* socketConnection) {
61 iot::mqtt::SocketContext* socketContext = nullptr;
62
63 mqtt::bridge::lib::Broker* broker = mqtt::bridge::lib::BridgeStore::instance().getBroker(socketConnection->getInstanceName());
64
65 if (broker != nullptr) {
66 VLOG(1) << " Creating Broker instance '" << broker->getName() << "' of Bridge '" << broker->getBridge().getName() << "'";
67 VLOG(1) << " Bridge client id : " << broker->getClientId();
68 VLOG(1) << " Transport: " << broker->getTransport();
69 VLOG(1) << " Protocol: " << broker->getProtocol();
70 VLOG(1) << " Encryption: " << broker->getEncryption();
71
72 VLOG(1) << " Topics:";
73 const std::list<iot::mqtt::Topic>& topics = broker->getTopics();
74 for (const iot::mqtt::Topic& topic : topics) {
75 VLOG(1) << " " << static_cast<uint16_t>(topic.getQoS()) << ":" << topic.getName();
76 }
77
78 socketContext =
79 new iot::mqtt::SocketContext(socketConnection, new mqtt::bridge::lib::Mqtt(socketConnection->getConnectionName(), *broker));
80 }
81
82 return socketContext;
83 }
