94 bool Mqtt::onSignal(int signum) {
95 VLOG(1) << "MQTT: On Exit due to '" << strsignal(signum) << "' (SIG" << utils::system::sigabbrev_np(signum) << " = " << signum
96 << ")";
97
98 sendDisconnect();
99
100 return Super::onSignal(signum);

103 void Mqtt::onConnack(const iot::mqtt::packets::Connack& connack) {
104 if (connack.getReturnCode() == 0) {
105 broker.getBridge().addMqtt(this);
106
107 sendSubscribe(broker.getTopics());
108 }
109 }
