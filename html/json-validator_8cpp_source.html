76{
77 const std::string id_;
78 std::weak_ptr<schema> target_;

83 {
84 auto target = target_.lock();

182 void insert(const json_uri &uri, const std::shared_ptr<schema> &s)
183 {
184 auto &file = get_or_create_file(uri.location());
185 auto sch = file.schemas.lower_bound(uri.fragment());
186 if (sch != file.schemas.end() && !(file.schemas.key_comp()(uri.fragment(), sch->first))) {
187 throw std::invalid_argument("schema with " + uri.to_string() + " already inserted");
188 return;
189 }
190
191 file.schemas.insert({uri.fragment(), s});
192
193 // was someone referencing this newly inserted schema?
194 auto unresolved = file.unresolved.find(uri.fragment());
195 if (unresolved != file.unresolved.end()) {
196 unresolved->second->set_target(s);
197 file.unresolved.erase(unresolved);
198 }
199 }
200
201 void insert_unknown_keyword(const json_uri &uri, const std::string &key, json &value)
202 {
203 auto &file = get_or_create_file(uri.location());
204 auto new_uri = uri.append(key);
205 auto fragment = new_uri.pointer();
206
207 // is there a reference looking for this unknown-keyword, which is thus no longer a unknown keyword but a schema
208 auto unresolved = file.unresolved.find(fragment.to_string());
209 if (unresolved != file.unresolved.end())
210 schema::make(value, this, {}, {{new_uri}});
211 else { // no, nothing ref'd it, keep for later
212
213 // need to create an object for each reference-token in the
214 // JSON-Pointer When not existing, a stringified integer reference
215 // token (e.g. "123") in the middle of the pointer will be
216 // interpreted a an array-index and an array will be created.
217
218 // json_pointer's reference_tokens is private - get them
219 std::deque<std::string> ref_tokens;
220 auto uri_pointer = uri.pointer();
221 while (!uri_pointer.empty()) {
222 ref_tokens.push_front(uri_pointer.back());
223 uri_pointer.pop_back();
224 }
225
226 // for each token create an object, if not already existing
227 auto unk_kw = &file.unknown_keywords;
228 for (auto &rt : ref_tokens) {
229 auto existing_object = unk_kw->find(rt);
230 if (existing_object == unk_kw->end())
231 (*unk_kw)[rt] = json::object();
232 unk_kw = &(*unk_kw)[rt];
233 }
234 (*unk_kw)[key] = value;
235 }
236
237 // recursively add possible subschemas of unknown keywords
238 if (value.type() == json::value_t::object)
239 for (auto &subsch : value.items())
240 insert_unknown_keyword(new_uri, subsch.key(), subsch.value());
241 }

243 std::shared_ptr<schema> get_or_create_ref(const json_uri &uri)
244 {
245 auto &file = get_or_create_file(uri.location());
246
247 // existing schema
248 auto sch = file.schemas.find(uri.fragment());
249 if (sch != file.schemas.end())
250 return sch->second;
251
252 // referencing an unknown keyword, turn it into schema
253 //
254 // an unknown keyword can only be referenced by a json-pointer,
255 // not by a plain name fragment
256 if (uri.pointer().to_string() != "") {
257 try {
258 auto &subschema = file.unknown_keywords.at(uri.pointer()); // null is returned if not existing
259 auto s = schema::make(subschema, this, {}, {{uri}}); // A JSON Schema MUST be an object or a boolean.
260 if (s) { // nullptr if invalid schema, e.g. null
261 file.unknown_keywords.erase(uri.fragment());
262 return s;
263 }
264 } catch (nlohmann::detail::out_of_range &) { // at() did not find it
265 }
266 }
267
268 // get or create a schema_ref
269 auto r = file.unresolved.lower_bound(uri.fragment());
270 if (r != file.unresolved.end() && !(file.unresolved.key_comp()(uri.fragment(), r->first))) {
271 return r->second; // unresolved, already seen previously - use existing reference
272 } else {
273 return file.unresolved.insert(r,
274 {uri.fragment(), std::make_shared<schema_ref>(uri.to_string(), this)})
275 ->second; // unresolved, create reference
276 }
277 }

279 void set_root_schema(json sch)
280 {
281 files_.clear();
282 root_ = schema::make(sch, this, {}, {{"#"}});
283
284 // load all files which have not yet been loaded
285 do {
286 bool new_schema_loaded = false;
287
288 // files_ is modified during parsing, iterators are invalidated
289 std::vector<std::string> locations;
290 for (auto &file : files_)
291 locations.push_back(file.first);
292
293 for (auto &loc : locations) {
294 if (files_[loc].schemas.size() == 0) { // nothing has been loaded for this file
295 if (loader_) {
296 json loaded_schema;
297
298 loader_(loc, loaded_schema);
299
300 schema::make(loaded_schema, this, {}, {{loc}});
301 new_schema_loaded = true;
302 } else {
303 throw std::invalid_argument("external schema reference '" + loc + "' needs loading, but no loader callback given");
304 }
305 }
306 }
307
308 if (!new_schema_loaded) // if no new schema loaded, no need to try again
309 break;
310 } while (1);
311
312 for (const auto &file : files_) {
313 if (file.second.unresolved.size() != 0) {
314 // Build a representation of the undefined
315 // references as a list of comma-separated strings.
316 auto n_urefs = file.second.unresolved.size();
317 std::string urefs = "[";
318
319 decltype(n_urefs) counter = 0;
320 for (const auto &p : file.second.unresolved) {
321 urefs += p.first;
322
323 if (counter != n_urefs - 1u) {
324 urefs += ", ";
325 }
326
327 ++counter;
328 }
329
330 urefs += "]";
331
332 throw std::invalid_argument("after all files have been parsed, '" +
333 (file.first == "" ? "<root>" : file.first) +
334 "' has still the following undefined references: " + urefs);
335 }
336 }
337 }
338
339 void validate(const json::json_pointer &ptr,
340 const json &instance,
341 json_patch &patch,
342 error_handler &e,
343 const json_uri &initial) const
344 {
345 if (!root_) {
346 e.error(ptr, "", "no root schema has yet been set for validating an instance");
347 return;
348 }
349
350 auto file_entry = files_.find(initial.location());
351 if (file_entry == files_.end()) {
352 e.error(ptr, "", "no file found serving requested root-URI. " + initial.location());
353 return;
354 }
355
356 auto &file = file_entry->second;
357 auto sch = file.schemas.find(initial.fragment());
358 if (sch == file.schemas.end()) {
359 e.error(ptr, "", "no schema find for request initial URI: " + initial.to_string());
360 return;
361 }
362
363 sch->second->validate(ptr, instance, patch, e);
364 }

433 void validate(const json::json_pointer &ptr, const json &instance, json_patch &patch, error_handler &e) const final

435 size_t count = 0;
436
437 for (auto &s : subschemata_) {
438 first_error_handler esub;

501class type_schema : public schema
502{
503 std::vector<std::shared_ptr<schema>> type_;
504 std::pair<bool, json> enum_, const_;
505 std::vector<std::shared_ptr<schema>> logic_;
506
507 static std::shared_ptr<schema> make(json &schema,
508 json::value_t type,
509 root_schema *,
510 const std::vector<nlohmann::json_uri> &,
511 std::set<std::string> &);
512
513 std::shared_ptr<schema> if_, then_, else_;
514
515 void validate(const json::json_pointer &ptr, const json &instance, json_patch &patch, error_handler &e) const override final
516 {
517 // depending on the type of instance run the type specific validator - if present
518 auto type = type_[static_cast<uint8_t>(instance.type())];
519
520 if (type)
521 type->validate(ptr, instance, patch, e);
522 else
523 e.error(ptr, instance, "unexpected instance type");
524
525 if (enum_.first) {
526 bool seen_in_enum = false;
527 for (auto &v : enum_.second)
528 if (instance == v) {
529 seen_in_enum = true;
530 break;
531 }
532
533 if (!seen_in_enum)
534 e.error(ptr, instance, "instance not found in required enum");
535 }
536
537 if (const_.first &&
538 const_.second != instance)
539 e.error(ptr, instance, "instance not const");
540
541 for (auto l : logic_)
542 l->validate(ptr, instance, patch, e);
543
544 if (if_) {
545 first_error_handler err;
546
547 if_->validate(ptr, instance, patch, err);
548 if (!err) {
549 if (then_)
550 then_->validate(ptr, instance, patch, e);
551 } else {
552 if (else_)
553 else_->validate(ptr, instance, patch, e);
554 }
555 }
556 if (instance.is_null()) {
557 patch.add(nlohmann::json::json_pointer{}, default_value_);
558 }
559 }
560
561protected:
562 virtual std::shared_ptr<schema> make_for_default_(
563 std::shared_ptr<::schema> & /* sch */,
564 root_schema * /* root */,
565 std::vector<nlohmann::json_uri> & /* uris */,
566 nlohmann::json &default_value) const override
567 {
568 auto result = std::make_shared<type_schema>(*this);
569 result->set_default_value(default_value);
570 return result;
571 };
572
573public:
574 type_schema(json &sch,
575 root_schema *root,
576 const std::vector<nlohmann::json_uri> &uris)
577 : schema(root), type_(static_cast<uint8_t>(json::value_t::discarded) + 1)
578 {
579 // association between JSON-schema-type and NLohmann-types
580 static const std::vector<std::pair<std::string, json::value_t>> schema_types = {
581 {"null", json::value_t::null},
582 {"object", json::value_t::object},
583 {"array", json::value_t::array},
584 {"string", json::value_t::string},
585 {"boolean", json::value_t::boolean},
586 {"integer", json::value_t::number_integer},
587 {"number", json::value_t::number_float},
588 };
589
590 std::set<std::string> known_keywords;
591
592 auto attr = sch.find("type");
593 if (attr == sch.end()) // no type field means all sub-types possible
594 for (auto &t : schema_types)
595 type_[static_cast<uint8_t>(t.second)] = type_schema::make(sch, t.second, root, uris, known_keywords);
596 else {
597 switch (attr.value().type()) { // "type": "type"
598
599 case json::value_t::string: {
600 auto schema_type = attr.value().get<std::string>();
601 for (auto &t : schema_types)
602 if (t.first == schema_type)
603 type_[static_cast<uint8_t>(t.second)] = type_schema::make(sch, t.second, root, uris, known_keywords);
604 } break;
605
606 case json::value_t::array: // "type": ["type1", "type2"]
607 for (auto &array_value : attr.value()) {
608 auto schema_type = array_value.get<std::string>();
609 for (auto &t : schema_types)
610 if (t.first == schema_type)
611 type_[static_cast<uint8_t>(t.second)] = type_schema::make(sch, t.second, root, uris, known_keywords);
612 }
613 break;
614
615 default:
616 break;
617 }
618
619 sch.erase(attr);
620 }
621
622 attr = sch.find("default");
623 if (attr != sch.end()) {
624 set_default_value(attr.value());
625 sch.erase(attr);
626 }
627
628 for (auto &key : known_keywords)
629 sch.erase(key);
630
631 // with nlohmann::json float instance (but number in schema-definition) can be seen as unsigned or integer -
632 // reuse the number-validator for integer values as well, if they have not been specified explicitly
633 if (type_[static_cast<uint8_t>(json::value_t::number_float)] && !type_[static_cast<uint8_t>(json::value_t::number_integer)])
634 type_[static_cast<uint8_t>(json::value_t::number_integer)] = type_[static_cast<uint8_t>(json::value_t::number_float)];
635
636 // #54: JSON-schema does not differentiate between unsigned and signed integer - nlohmann::json does
637 // we stick with JSON-schema: use the integer-validator if instance-value is unsigned
638 type_[static_cast<uint8_t>(json::value_t::number_unsigned)] = type_[static_cast<uint8_t>(json::value_t::number_integer)];
639
640 // special for binary types
641 if (type_[static_cast<uint8_t>(json::value_t::string)]) {
642 type_[static_cast<uint8_t>(json::value_t::binary)] = type_[static_cast<uint8_t>(json::value_t::string)];
643 }
644
645 attr = sch.find("enum");
646 if (attr != sch.end()) {
647 enum_ = {true, attr.value()};
648 sch.erase(attr);
649 }
650
651 attr = sch.find("const");
652 if (attr != sch.end()) {
653 const_ = {true, attr.value()};
654 sch.erase(attr);
655 }
656
657 attr = sch.find("not");
658 if (attr != sch.end()) {
659 logic_.push_back(std::make_shared<logical_not>(attr.value(), root, uris));
660 sch.erase(attr);
661 }
662
663 attr = sch.find("allOf");
664 if (attr != sch.end()) {
665 logic_.push_back(std::make_shared<logical_combination<allOf>>(attr.value(), root, uris));
666 sch.erase(attr);
667 }
668
669 attr = sch.find("anyOf");
670 if (attr != sch.end()) {
671 logic_.push_back(std::make_shared<logical_combination<anyOf>>(attr.value(), root, uris));
672 sch.erase(attr);
673 }
674
675 attr = sch.find("oneOf");
676 if (attr != sch.end()) {
677 logic_.push_back(std::make_shared<logical_combination<oneOf>>(attr.value(), root, uris));
678 sch.erase(attr);
679 }
680
681 attr = sch.find("if");
682 if (attr != sch.end()) {
683 auto attr_then = sch.find("then");
684 auto attr_else = sch.find("else");
685
686 if (attr_then != sch.end() || attr_else != sch.end()) {
687 if_ = schema::make(attr.value(), root, {"if"}, uris);
688
689 if (attr_then != sch.end()) {
690 then_ = schema::make(attr_then.value(), root, {"then"}, uris);
691 sch.erase(attr_then);
692 }
693
694 if (attr_else != sch.end()) {
695 else_ = schema::make(attr_else.value(), root, {"else"}, uris);
696 sch.erase(attr_else);
697 }
698 }
699 sch.erase(attr);
700 }
701 }

704class string : public schema
705{
706 std::pair<bool, size_t> maxLength_{false, 0};
707 std::pair<bool, size_t> minLength_{false, 0};
708
709#ifndef NO_STD_REGEX
710 std::pair<bool, REGEX_NAMESPACE::regex> pattern_{false, REGEX_NAMESPACE::regex()};
711 std::string patternString_;
712#endif
713
714 std::pair<bool, std::string> format_;
715 std::tuple<bool, std::string, std::string> content_{false, "", ""};
716
717 std::size_t utf8_length(const std::string &s) const
718 {
719 size_t len = 0;
720 for (auto c : s)
721 if ((c & 0xc0) != 0x80)
722 len++;
723 return len;
724 }
725
726 void validate(const json::json_pointer &ptr, const json &instance, json_patch &, error_handler &e) const override
727 {
728 if (minLength_.first) {
729 if (utf8_length(instance.get<std::string>()) < minLength_.second) {
730 std::ostringstream s;
731 s << "instance is too short as per minLength:" << minLength_.second;
732 e.error(ptr, instance, s.str());
733 }
734 }
735
736 if (maxLength_.first) {
737 if (utf8_length(instance.get<std::string>()) > maxLength_.second) {
738 std::ostringstream s;
739 s << "instance is too long as per maxLength: " << maxLength_.second;
740 e.error(ptr, instance, s.str());
741 }
742 }
743
744 if (std::get<0>(content_)) {
745 if (root_->content_check() == nullptr)
746 e.error(ptr, instance, std::string("a content checker was not provided but a contentEncoding or contentMediaType for this string have been present: '") + std::get<1>(content_) + "' '" + std::get<2>(content_) + "'");
747 else {
748 try {
749 root_->content_check()(std::get<1>(content_), std::get<2>(content_), instance);
750 } catch (const std::exception &ex) {
751 e.error(ptr, instance, std::string("content-checking failed: ") + ex.what());
752 }
753 }
754 } else if (instance.type() == json::value_t::binary) {
755 e.error(ptr, instance, "expected string, but get binary data");
756 }
757
758 if (instance.type() != json::value_t::string) {
759 return; // next checks only for strings
760 }
761
762#ifndef NO_STD_REGEX
763 if (pattern_.first &&
764 !REGEX_NAMESPACE::regex_search(instance.get<std::string>(), pattern_.second))
765 e.error(ptr, instance, "instance does not match regex pattern: " + patternString_);
766#endif
767
768 if (format_.first) {
769 if (root_->format_check() == nullptr)
770 e.error(ptr, instance, std::string("a format checker was not provided but a format keyword for this string is present: ") + format_.second);
771 else {
772 try {
773 root_->format_check()(format_.second, instance.get<std::string>());
774 } catch (const std::exception &ex) {
775 e.error(ptr, instance, std::string("format-checking failed: ") + ex.what());
776 }
777 }
778 }
779 }

788 sch.erase(attr);
789 }
790
791 attr = sch.find("minLength");

826 }

842
843 format_ = {true, attr.value().get<std::string>()};
844 sch.erase(attr);
845 }

849template <typename T>
850class numeric : public schema
851{
852 std::pair<bool, T> maximum_{false, 0};
853 std::pair<bool, T> minimum_{false, 0};
854
855 bool exclusiveMaximum_ = false;
856 bool exclusiveMinimum_ = false;
857
858 std::pair<bool, json::number_float_t> multipleOf_{false, 0};
859
860 // multipleOf - if the remainder of the division is 0 -> OK
861 bool violates_multiple_of(T x) const
862 {
863 double res = std::remainder(x, multipleOf_.second);
864 double eps = std::nextafter(x, 0) - static_cast<double>(x);
865 return std::fabs(res) > std::fabs(eps);
866 }
867
868 void validate(const json::json_pointer &ptr, const json &instance, json_patch &, error_handler &e) const override
869 {
870 T value = instance; // conversion of json to value_type
871
872 if (multipleOf_.first && value != 0) // zero is multiple of everything
873 if (violates_multiple_of(value))
874 e.error(ptr, instance, "instance is not a multiple of " + std::to_string(multipleOf_.second));
875
876 if (maximum_.first) {
877 if (exclusiveMaximum_ && value >= maximum_.second)
878 e.error(ptr, instance, "instance exceeds or equals maximum of " + std::to_string(maximum_.second));
879 else if (value > maximum_.second)
880 e.error(ptr, instance, "instance exceeds maximum of " + std::to_string(maximum_.second));
881 }
882
883 if (minimum_.first) {
884 if (exclusiveMinimum_ && value <= minimum_.second)
885 e.error(ptr, instance, "instance is below or equals minimum of " + std::to_string(minimum_.second));
886 else if (value < minimum_.second)
887 e.error(ptr, instance, "instance is below minimum of " + std::to_string(minimum_.second));
888 }
889 }

891public:

893 : schema(root)

904 kw.insert("minimum");
905 }

907 attr = sch.find("exclusiveMaximum");
908 if (attr != sch.end()) {
909 exclusiveMaximum_ = true;

912 }

914 attr = sch.find("exclusiveMinimum");
915 if (attr != sch.end()) {
916 exclusiveMinimum_ = true;

990class object : public schema
991{
992 std::pair<bool, size_t> maxProperties_{false, 0};
993 std::pair<bool, size_t> minProperties_{false, 0};
994 std::vector<std::string> required_;
995
996 std::map<std::string, std::shared_ptr<schema>> properties_;
997#ifndef NO_STD_REGEX
998 std::vector<std::pair<REGEX_NAMESPACE::regex, std::shared_ptr<schema>>> patternProperties_;
999#endif
1000 std::shared_ptr<schema> additionalProperties_;
1001
1002 std::map<std::string, std::shared_ptr<schema>> dependencies_;
1003
1004 std::shared_ptr<schema> propertyNames_;
1005
1006 void validate(const json::json_pointer &ptr, const json &instance, json_patch &patch, error_handler &e) const override
1007 {
1008 if (maxProperties_.first && instance.size() > maxProperties_.second)
1009 e.error(ptr, instance, "too many properties");
1010
1011 if (minProperties_.first && instance.size() < minProperties_.second)
1012 e.error(ptr, instance, "too few properties");
1013
1014 for (auto &r : required_)
1015 if (instance.find(r) == instance.end())
1016 e.error(ptr, instance, "required property '" + r + "' not found in object");
1017
1018 // for each property in instance
1019 for (auto &p : instance.items()) {
1020 if (propertyNames_)
1021 propertyNames_->validate(ptr, p.key(), patch, e);
1022
1023 bool a_prop_or_pattern_matched = false;
1024 auto schema_p = properties_.find(p.key());
1025 // check if it is in "properties"
1026 if (schema_p != properties_.end()) {
1027 a_prop_or_pattern_matched = true;
1028 schema_p->second->validate(ptr / p.key(), p.value(), patch, e);
1029 }
1030
1031#ifndef NO_STD_REGEX
1032 // check all matching patternProperties
1033 for (auto &schema_pp : patternProperties_)
1034 if (REGEX_NAMESPACE::regex_search(p.key(), schema_pp.first)) {
1035 a_prop_or_pattern_matched = true;
1036 schema_pp.second->validate(ptr / p.key(), p.value(), patch, e);
1037 }
1038#endif
1039
1040 // check additionalProperties as a last resort
1041 if (!a_prop_or_pattern_matched && additionalProperties_) {
1042 first_error_handler additional_prop_err;
1043 additionalProperties_->validate(ptr / p.key(), p.value(), patch, additional_prop_err);
1044 if (additional_prop_err)
1045 e.error(ptr, instance, "validation failed for additional property '" + p.key() + "': " + additional_prop_err.message_);
1046 }
1047 }
1048
1049 // reverse search
1050 for (auto const &prop : properties_) {
1051 const auto finding = instance.find(prop.first);
1052 if (instance.end() == finding) { // if the prop is not in the instance
1053 const auto &default_value = prop.second->default_value(ptr, instance, e);
1054 if (!default_value.is_null()) { // if default value is available
1055 patch.add((ptr / prop.first), default_value);
1056 }
1057 }
1058 }
1059
1060 for (auto &dep : dependencies_) {
1061 auto prop = instance.find(dep.first);
1062 if (prop != instance.end()) // if dependency-property is present in instance
1063 dep.second->validate(ptr / dep.first, instance, patch, e); // validate
1064 }
1065 }
1066
1067public:
1068 object(json &sch,
1069 root_schema *root,
1070 const std::vector<nlohmann::json_uri> &uris)
1071 : schema(root)
1072 {
1073 auto attr = sch.find("maxProperties");
1074 if (attr != sch.end()) {
1075 maxProperties_ = {true, attr.value().get<size_t>()};
1076 sch.erase(attr);
1077 }
1078
1079 attr = sch.find("minProperties");
1080 if (attr != sch.end()) {
1081 minProperties_ = {true, attr.value().get<size_t>()};
1082 sch.erase(attr);
1083 }
1084
1085 attr = sch.find("required");
1086 if (attr != sch.end()) {
1087 required_ = attr.value().get<std::vector<std::string>>();
1088 sch.erase(attr);
1089 }
1090
1091 attr = sch.find("properties");
1092 if (attr != sch.end()) {
1093 for (auto prop : attr.value().items())
1094 properties_.insert(
1095 std::make_pair(
1096 prop.key(),
1097 schema::make(prop.value(), root, {"properties", prop.key()}, uris)));
1098 sch.erase(attr);
1099 }
1100
1101#ifndef NO_STD_REGEX
1102 attr = sch.find("patternProperties");
1103 if (attr != sch.end()) {
1104 for (auto prop : attr.value().items())
1105 patternProperties_.push_back(
1106 std::make_pair(
1107 REGEX_NAMESPACE::regex(prop.key(), REGEX_NAMESPACE::regex::ECMAScript),
1108 schema::make(prop.value(), root, {prop.key()}, uris)));
1109 sch.erase(attr);
1110 }
1111#endif
1112
1113 attr = sch.find("additionalProperties");
1114 if (attr != sch.end()) {
1115 additionalProperties_ = schema::make(attr.value(), root, {"additionalProperties"}, uris);
1116 sch.erase(attr);
1117 }
1118
1119 attr = sch.find("dependencies");
1120 if (attr != sch.end()) {
1121 for (auto &dep : attr.value().items())
1122 switch (dep.value().type()) {
1123 case json::value_t::array:
1124 dependencies_.emplace(dep.key(),
1125 std::make_shared<required>(
1126 dep.value().get<std::vector<std::string>>(), root));
1127 break;
1128
1129 default:
1130 dependencies_.emplace(dep.key(),
1131 schema::make(dep.value(), root, {"dependencies", dep.key()}, uris));
1132 break;
1133 }
1134 sch.erase(attr);
1135 }
1136
1137 attr = sch.find("propertyNames");
1138 if (attr != sch.end()) {
1139 propertyNames_ = schema::make(attr.value(), root, {"propertyNames"}, uris);
1140 sch.erase(attr);
1141 }
1142
1143 attr = sch.find("default");
1144 if (attr != sch.end()) {
1145 set_default_value(*attr);
1146 }
1147 }

1150class array : public schema
1151{
1152 std::pair<bool, size_t> maxItems_{false, 0};
1153 std::pair<bool, size_t> minItems_{false, 0};
1154 bool uniqueItems_ = false;
1155
1156 std::shared_ptr<schema> items_schema_;
1157
1158 std::vector<std::shared_ptr<schema>> items_;
1159 std::shared_ptr<schema> additionalItems_;
1160
1161 std::shared_ptr<schema> contains_;
1162
1163 void validate(const json::json_pointer &ptr, const json &instance, json_patch &patch, error_handler &e) const override
1164 {
1165 if (maxItems_.first && instance.size() > maxItems_.second)
1166 e.error(ptr, instance, "array has too many items");
1167
1168 if (minItems_.first && instance.size() < minItems_.second)
1169 e.error(ptr, instance, "array has too few items");
1170
1171 if (uniqueItems_) {
1172 for (auto it = instance.cbegin(); it != instance.cend(); ++it) {
1173 auto v = std::find(it + 1, instance.end(), *it);
1174 if (v != instance.end())
1175 e.error(ptr, instance, "items have to be unique for this array");
1176 }
1177 }
1178
1179 size_t index = 0;
1180 if (items_schema_)
1181 for (auto &i : instance) {
1182 items_schema_->validate(ptr / index, i, patch, e);
1183 index++;
1184 }
1185 else {
1186 auto item = items_.cbegin();
1187 for (auto &i : instance) {
1188 std::shared_ptr<schema> item_validator;
1189 if (item == items_.cend())
1190 item_validator = additionalItems_;
1191 else {
1192 item_validator = *item;
1193 item++;
1194 }
1195
1196 if (!item_validator)
1197 break;
1198
1199 item_validator->validate(ptr / index, i, patch, e);
1200 }
1201 }
1202
1203 if (contains_) {
1204 bool contained = false;
1205 for (auto &item : instance) {
1206 first_error_handler local_e;
1207 contains_->validate(ptr, item, patch, local_e);
1208 if (!local_e) {
1209 contained = true;
1210 break;
1211 }
1212 }
1213 if (!contained)
1214 e.error(ptr, instance, "array does not contain required element as per 'contains'");
1215 }
1216 }

1218public:
1219 array(json &sch, root_schema *root, const std::vector<nlohmann::json_uri> &uris)
1220 : schema(root)
1221 {
1222 auto attr = sch.find("maxItems");
1223 if (attr != sch.end()) {
1224 maxItems_ = {true, attr.value().get<size_t>()};
1225 sch.erase(attr);
1226 }
1227
1228 attr = sch.find("minItems");
1229 if (attr != sch.end()) {
1230 minItems_ = {true, attr.value().get<size_t>()};
1231 sch.erase(attr);
1232 }
1233
1234 attr = sch.find("uniqueItems");
1235 if (attr != sch.end()) {
1236 uniqueItems_ = attr.value().get<bool>();
1237 sch.erase(attr);
1238 }
1239
1240 attr = sch.find("items");
1241 if (attr != sch.end()) {
1242
1243 if (attr.value().type() == json::value_t::array) {
1244 size_t c = 0;
1245 for (auto &subsch : attr.value())
1246 items_.push_back(schema::make(subsch, root, {"items", std::to_string(c++)}, uris));

1257
1258 sch.erase(attr);
1259 }
1260
1261 attr = sch.find("contains");
1262 if (attr != sch.end()) {
1263 contains_ = schema::make(attr.value(), root, {"contains"}, uris);
1264 sch.erase(attr);
1265 }

1270 json::value_t type,
1271 root_schema *root,
1272 const std::vector<nlohmann::json_uri> &uris,
1273 std::set<std::string> &kw)
1274{
1275 switch (type) {
1276 case json::value_t::null:

1278
1279 case json::value_t::number_unsigned:
1280 case json::value_t::number_integer:

1282 case json::value_t::number_float:
1283 return std::make_shared<numeric<json::number_float_t>>(schema, root, kw);
1284 case json::value_t::string:
1285 return std::make_shared<string>(schema, root);
1286 case json::value_t::boolean:
1287 return std::make_shared<boolean_type>(schema, root);
1288 case json::value_t::object:

1294 break;
1295
1296 case json::value_t::binary:
1297 break;
1298 }
1299 return nullptr;

1306std::shared_ptr<schema> schema::make(json &schema,
1307 root_schema *root,
1308 const std::vector<std::string> &keys,
1309 std::vector<nlohmann::json_uri> uris)
1310{
1311 // remove URIs which contain plain name identifiers, as sub-schemas cannot be referenced
1312 for (auto uri = uris.begin(); uri != uris.end();)
1313 if (uri->identifier() != "")
1314 uri = uris.erase(uri);
1315 else
1316 uri++;
1317
1318 // append to all URIs the keys for this sub-schema
1319 for (auto &key : keys)
1320 for (auto &uri : uris)
1321 uri = uri.append(key);
1322
1323 std::shared_ptr<::schema> sch;
1324
1325 // boolean schema
1326 if (schema.type() == json::value_t::boolean)
1327 sch = std::make_shared<boolean>(schema, root);
1328 else if (schema.type() == json::value_t::object) {
1329
1330 auto attr = schema.find("$id"); // if $id is present, this schema can be referenced by this ID
1331 // as an additional URI
1332 if (attr != schema.end()) {
1333 if (std::find(uris.begin(),
1334 uris.end(),
1335 attr.value().get<std::string>()) == uris.end())
1336 uris.push_back(uris.back().derive(attr.value().get<std::string>())); // so add it to the list if it is not there already
1337 schema.erase(attr);
1338 }
1339
1340 attr = schema.find("definitions");
1341 if (attr != schema.end()) {
1342 for (auto &def : attr.value().items())
1343 schema::make(def.value(), root, {"definitions", def.key()}, uris);
1344 schema.erase(attr);
1345 }
1346
1347 attr = schema.find("$ref");
1348 if (attr != schema.end()) { // this schema is a reference
1349 // the last one on the uri-stack is the last id seen before coming here,
1350 // so this is the origial URI for this reference, the $ref-value has thus be resolved from it
1351 auto id = uris.back().derive(attr.value().get<std::string>());
1352 sch = root->get_or_create_ref(id);
1353
1354 schema.erase(attr);
1355
1356 // special case where we break draft-7 and allow overriding of properties when a $ref is used
1357 attr = schema.find("default");
1358 if (attr != schema.end()) {
1359 // copy the referenced schema depending on the underlying type and modify the default value
1360 if (auto new_sch = sch->make_for_default_(sch, root, uris, attr.value())) {
1361 sch = new_sch;
1362 }
1363 schema.erase(attr);
1364 }
1365 } else {
1366 sch = std::make_shared<type_schema>(schema, root, uris);
1367 }
1368
1369 schema.erase("$schema");
1370 schema.erase("title");
1371 schema.erase("description");
1372 } else {
1373 throw std::invalid_argument("invalid JSON-type for a schema for " + uris[0].to_string() + ", expected: boolean or object");
1374 }
1375
1376 for (auto &uri : uris) { // for all URIs this schema is referenced by
1377 root->insert(uri, sch);
1378
1379 if (schema.type() == json::value_t::object)
1380 for (auto &u : schema.items())
1381 root->insert_unknown_keyword(uri, u.key(), u.value()); // insert unknown keywords for later reference
1382 }
1383 return sch;
1384}
